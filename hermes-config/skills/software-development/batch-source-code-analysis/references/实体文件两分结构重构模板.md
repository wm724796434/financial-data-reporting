# 实体文件两分结构重构模板

> 本文件记录批量重构实体文件为两分结构的完整 Python 脚本模式和边缘情况处理。
> 对应 skill 中"增强工作流：实体文件两分结构设计"章节。

## 完整模板脚本

```python
import os, re

ENTITY_DIR = "实体"
ORIG_DIR = "参考资料/金融基础数据系统"
SRC_DIR = "源码解析/加工层存储"

def parse_orig_content(orig_path):
    """从原文文件提取结构化数据"""
    result = {
        'scope': '',          # 官方报送范围
        'inclusion': [],      # 纳入口径
        'exclusion': [],      # 排除口径
        'biz_elements': '',   # 关键业务要素
        'field_count': 0,     # 字段数量
        'time_range': '',     # 时间范围
        'granularity': ''     # 报送粒度
    }
    if not os.path.exists(orig_path):
        return result

    with open(orig_path, 'r', encoding='utf-8') as f:
        content = f.read()

    # 官方报送范围
    m = re.search(r'### 1\.\s*官方报送范围\s*\n(.*?)(?=\n###|\Z)', content, re.DOTALL)
    if m: result['scope'] = m.group(1).strip()

    # 纳入与排除口径
    m = re.search(r'### 7\.\s*纳入与排除口径\s*\n(.*?)(?=\n###|\Z)', content, re.DOTALL)
    if m:
        for line in m.group(1).split('\n'):
            line = line.strip()
            if '纳入口径' in line:
                result['inclusion'].append(line.split('纳入口径')[1].strip().strip('：:').strip())
            elif '排除口径' in line:
                result['exclusion'].append(line.split('排除口径')[1].strip().strip('：:').strip())

    # 关键业务要素
    m = re.search(r'### 6\.\s*关键业务要素\s*\n(.*?)(?=\n###|\Z)', content, re.DOTALL)
    if m: result['biz_elements'] = m.group(1).strip()

    # 字段数量
    result['field_count'] = len(re.findall(r'^\|\s*\d+\s*\|', content, re.MULTILINE))

    # 时间范围
    m = re.search(r'### 4\.\s*时间范围\s*\n\s*(.*?)(?=\n###|\Z)', content, re.DOTALL)
    if m: result['time_range'] = m.group(1).strip()

    # 报送粒度
    m = re.search(r'### 5\.\s*报送粒度\s*\n\s*(.*?)(?=\n###|\Z)', content, re.DOTALL)
    if m: result['granularity'] = m.group(1).strip()

    return result


def extract_prc_info(iface):
    """从PRC源码头提取修改历史和集市表引用"""
    result = {'has_prc': False, 'prc_name': '', 'history': ''}
    if not iface:
        return result
    iface_lower = iface.lower()
    prc_name = f"bsp_sp_{iface_lower}.prc"
    prc_path = os.path.join("源码解析/加工层存储", prc_name)
    if not os.path.exists(prc_path):
        return result
    result['has_prc'] = True
    result['prc_name'] = prc_name
    with open(prc_path, 'r', encoding='utf-8') as f:
        header = f.read(2000)
    m = re.search(r'-- 修改历史\s*\n(.*?)(?=--\s|\Z)', header, re.DOTALL)
    if m: result['history'] = m.group(1).strip()
    return result


def build_new_entity(entity_name, iface, chapter, orig_basename, orig_data, old_content):
    """按两分结构模板生成实体文件"""
    src = extract_prc_info(iface)
    lines = []

    # Title
    if iface:
        lines.append(f'# {entity_name}（{iface}）')
    else:
        lines.append(f'# {entity_name}')
    lines.append('')
    lines.append(f'> **来源**：[{orig_basename}](../参考资料/金融基础数据系统/{orig_basename})')
    lines.append('')

    # 业务分类
    lines.append('## 业务分类')
    lines.append('')
    lines.append('| 属性 | 值 |')
    lines.append('|------|-----|')
    if iface:
        lines.append(f'| 接口表 | `{iface}` |')
    else:
        lines.append(f'| 接口表 | （无对应接口表） |')
    lines.append(f'| 中文名 | {entity_name} |')
    if chapter:
        lines.append(f'| 章节 | {chapter} |')
    if orig_data['time_range']:
        lines.append(f'| 时间范围 | {orig_data["time_range"]} |')
    if orig_data['granularity']:
        lines.append(f'| 报送粒度 | {orig_data["granularity"]} |')
    lines.append('')
    lines.append('---')
    lines.append('')

    # PART 1
    lines.append('# 第一部分：发文原文要求（人行规范层）')
    lines.append('')
    lines.append('> **用于回答"这个表要求报送什么"、"新增业务应该报到哪个表"等问题**')
    lines.append('')
    lines.append('## 1. 官方报送范围')
    lines.append('')
    lines.append(orig_data['scope'] or '（请查看原文文件）')
    lines.append('')
    lines.append('## 2. 纳入与排除口径')
    lines.append('')
    if orig_data['inclusion'] or orig_data['exclusion']:
        lines.append('| 方向 | 内容 |')
        lines.append('|------|------|')
        for inc in orig_data['inclusion']:
            lines.append(f'| ✅ 纳入 | {inc} |')
        for exc in orig_data['exclusion']:
            lines.append(f'| ❌ 排除 | {exc} |')
    else:
        lines.append('（请查看原文文件）')
    lines.append('')
    lines.append('## 3. 关键业务要素（要求报送的内容维度）')
    lines.append('')
    lines.append(orig_data['biz_elements'] or '（请查看原文文件）')
    lines.append('')
    lines.append('## 4. 字段清单')
    lines.append('')
    lines.append(f'完整字段清单（共{orig_data["field_count"]}个字段）及填报说明、码值表详见原文文件：')
    lines.append(f'[{orig_basename}](../参考资料/金融基础数据系统/{orig_basename})')
    lines.append('')
    lines.append('---')
    lines.append('')

    # PART 2
    lines.append('# 第二部分：代码取数业务范围（实现层）')
    lines.append('')
    lines.append('> **用于回答"这个表怎么取数"、"取了哪些业务"、"业务变更对金数有什么影响"等问题**')
    lines.append('')
    lines.append('## 1. 数据生成程序')
    lines.append('')
    lines.append(f'`{src["prc_name"]}`（加工层存储过程）' if src['has_prc'] else '（当前实体暂无对应的加工层存储过程）')
    lines.append('')

    # 监管集市表 - 从旧内容提取
    m = re.search(r'## 引用的监管集市表\s*\n(.*?)(?=\n## |\Z)', old_content, re.DOTALL)
    lines.append('## 2. 监管集市表来源')
    lines.append('')
    lines.append(m.group(1).strip() if m else '（当前实体暂无明确的监管集市表依赖记录）')
    lines.append('')

    # 关键逻辑/业务筛选条件
    lines.append('## 3. 业务筛选条件')
    lines.append('')
    m = re.search(r'## 关键逻辑\s*\n(.*?)(?=\n## |\Z)', old_content, re.DOTALL)
    if m:
        logic = re.sub(r'^- \*\*功能\*\*.*?\n', '', m.group(1).strip())
        lines.append(logic)
    else:
        lines.append('（请查看源码解析文件获取详细取数逻辑）')
    lines.append('')

    # 特殊处理规则
    lines.append('## 4. 特殊处理规则')
    lines.append('')
    m = re.search(r'### 特殊处理规则\s*\n(.*?)(?=\n## |\Z)', old_content, re.DOTALL)
    lines.append(m.group(1).strip() if m else '（当前实体暂无特殊处理规则）')
    lines.append('')

    # 历史变更
    lines.append('## 5. 历史变更记录')
    lines.append('')
    if src['history']:
        lines.append(f'```\n{src["history"]}\n```')
    else:
        lines.append('（请查看源码文件头部注释获取详细修改历史）')
    lines.append('')

    # 涉及源码文件
    lines.append('## 6. 涉及源码文件')
    lines.append('')
    m = re.search(r'## 涉及源码文件\s*\n(.*?)(?=\n## |\Z)', old_content, re.DOTALL)
    lines.append(m.group(1).strip() if m else '（当前实体暂无明确的源码文件关联记录）')
    lines.append('')

    # Footer
    lines.append('---')
    lines.append('')
    lines.append('> **使用说明**：')
    lines.append('> - 问"**要求报送什么**"→ 查**第一部分（发文原文要求）**')
    lines.append('> - 问"**怎么取数/取了哪些业务/业务变更影响**"→ 查**第二部分（代码取数业务范围）**')
    lines.append('')
    lines.append(f'*← [返回首页索引](../首页/index.md) · 原文：[{orig_basename}](../参考资料/金融基础数据系统/{orig_basename})*')

    return '\n'.join(lines)
```

## 已知边缘情况处理

### 1. 五篇大文章无接口表代码
这些文件在旧版本中标题为 `# 五篇大文章-3.X标题`（无括号接口表名），脚本提取正则 `^#\s*(.+?)(?:（(.+?)）)` 匹配失败，导致产生空括号 `（）（）`。

**修复方式**：
```python
# 读取旧实体文件后，检查标题是否正确
title_match = re.search(r'^#\s*(.+?)(?:（(.+?)）)?\s*$', content, re.MULTILINE)
if not title_match or (title_match and not title_match.group(2)):
    # 五篇大文章特殊处理：不写接口表名
    pass
```

### 2. 原文引用路径重复
旧实体文件中 `来源` 行格式为 `> **来源**：[xxx](../参考资料/...xxx.md)`。
用正则 `\]\(\.\./(参考资料/金融基础数据系统/[^)]+)\)` 提取时，group(1) 为 `参考资料/...`（无 `../` 前缀）。

**修复方式**：一律用 `os.path.basename(orig_ref)` 只保留文件名，模板中重新拼入 `../参考资料/金融基础数据系统/` 前缀。

### 3. 字段计数正则歧义
原文文件中字段清单表格的 `| 序号 |` 行：用 `re.findall(r'^\|\s*\d+\s*\|', content, re.MULTILINE)`。
注意：如果原文文件字段表后有其他带数字的表格，会误计。可在匹配前截取到"处理记录"之前的内容。

### 4. 纳入排除口径提取不完整
部分原文文件将纳入排除写在单一段落而非表格中，用 `纳入`/`排除` 关键词切割。
```python
if '纳入' in line and '口径' in line:
    result['inclusion'].append(line.split('纳入口径')[1]...)
```
如果一行包含多个条件，需按 `；` 或 `。` 拆分。

### 5. 破损链接残留（五篇大文章特有）

当 `iface` 为空字符串时，`orig_basename` 的 fallback 逻辑 `f'{iface}_原文.md'` 会产生 `_原文.md`，再经过 Python 字符串拼接会演变为 `）（_原文.md` 的破损链接。该模式会在文件的三个位置出现：来源行、字段清单区原文链接、页脚原文链接。

**修复方式**（全局字符串替换）：
```bash
cd 实体
for ef in 五篇大文章-3.*.md; do
    base="${ef%.md}"
    orig_file="${base}-原文.md"
    python3 -c "
with open('$ef', 'r', encoding='utf-8') as f:
    content = f.read()
content = content.replace('）（_原文.md', '${orig_file}')
with open('$ef', 'w', encoding='utf-8') as f:
    f.write(content)
"
done
```

**预防措施**：在提取元信息后，对五篇大文章增加显式的 orig_basename 映射：
```python
# 在 extract 元信息后
if not iface and '五篇大文章' in entity_name:
    # 五篇大文章的原文文件名 = 实体文件名去掉.md + -原文.md
    orig_basename = ef.replace('.md', '') + '-原文.md'
```

## 执行建议

1. **先在单个文件上测试**：选一个有代表性的实体文件（如存量单位贷款），确认模板生成正确
2. **用 terminal 运行 Python 脚本**：放在 `<< 'PYEOF' ... PYEOF` 中执行
3. **分批处理**：如果文件量大（50+），分批运行避免单次超时
4. **验证**：处理后检查三点——路径无重复、链接无破损、标题无空括号
5. **修复残留**：用 `grep -rn '重复路径' 实体/*.md` 和 `grep -rn '破损链接' 实体/*.md` 检查

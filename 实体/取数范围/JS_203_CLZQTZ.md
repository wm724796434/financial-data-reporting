# JS_203_CLZQTZ — 取数范围

> 来源程序: `bsp_sp_js_203_clzqtz.prc`

## 数据源表（FROM）

- SMTMODS.L_AGRE_BOND_INFO AS B
- SMTMODS.L_CUST_ALL AS D
- SMTMODS.L_CUST_EXTERNAL_INFO AS C
- SMTMODS.L_CUST_EXTERNAL_INFO AS C2
- SMTMODS.L_PUBL_RATE AS T3

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_AGRE_BOND_INFO AS B ON REPLACE(A.ACCT_NUM, '0041800014', '041800014') = B.STOCK_CD AND A.DATA_DATE = B.DATA_DATE AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_EXTERNAL_INFO AS C ON A.CUST_ID = C.CUST_ID AND A.DATA_DATE = C.DATA_DATE AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_EXTERNAL_INFO AS C2 ON (CASE WHEN C.ORG_FULLNAME LIKE '%省政府' THEN REPLACE(C.ORG_FULLNAME, '省政府', '省财政厅') WHEN C.ORG_FULLNAME LIKE '%自治区政府' THEN REPLACE(C.ORG_FULLNAME, '自治区政府', '自治区财政厅') WHEN C.ORG_FULLNAME LIKE '%市人民政府' THEN REPLACE(C.ORG_FULLNAME, '市人民政府', '市财政局') WHEN C.ORG_FULLNAME LIKE '%市政府' THEN REPLACE(C.ORG_FULLNAME, '市政府', '市财政局') WHEN C.ORG_FULLNAME LIKE '%省人民政府' THEN REPLACE(C.ORG_FULLNAME, '省人民政府', '省财政厅') WHEN C.ORG_FULLNAME LIKE '%自治区人民政府' THEN REPLACE(C.ORG_FULLNAME, '自治区人民政府', '自治区财政厅') END) = C2.ORG_FULLNAME AND A.DATA_DATE = C2.DATA_DATE AND C2.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS D ON A.CUST_ID = D.CUST_ID AND A.DATA_DATE = D.DATA_DATE AND D.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS T3 ON T3.BASIC_CCY = A.CURR_CD AND T3.FORWARD_CCY = 'CNY' AND T3.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_203_CLZQTZ'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `A.DATA_DATE = IS_DATE`
4. `A.GL_ITEM_CODE IN ('15030103', '11010101', '15010103', '15030101', '11010103', '11010102', '15030102', '15010101', '15010102')`
5. `a.DATE_SOURCESD = '债券投资'`
6. `A.PRINCIPAL_BALANCE > 0`

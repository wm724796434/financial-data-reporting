# 债券投资发生额信息_JS_203_ZQTZFS — 取数范围

> 来源程序: `bsp_sp_js_203_zqtzfs.prc`

## 数据源表（FROM）

- SMTMODS.L_ACCT_FUND_INVEST AS M
- SMTMODS.L_AGRE_BOND_INFO AS B
- SMTMODS.L_CUST_ALL AS D
- SMTMODS.L_CUST_EXTERNAL_INFO AS C
- SMTMODS.L_CUST_EXTERNAL_INFO AS C2
- SMTMODS.L_PUBL_RATE AS T3

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_FUND_INVEST AS M ON SUBSTRING(A.CONTRACT_NUM, 0, LENGTH(A.CONTRACT_NUM) - 1) = REPLACE(M.ACCT_NUM, '0041800014', '041800014') AND M.DATA_DATE = IS_DATE AND M.GL_ITEM_CODE IN ('15030103', '11010101', '15010103', '15030101', '11010103', '11010102', '15030102', '15010101', '15010102') AND M.DATE_SOURCESD = '债券投资' AND A.ACCT_NO = M.ACCT_NO
- INNER JOIN SMTMODS.L_AGRE_BOND_INFO AS B ON SUBSTRING(A.CONTRACT_NUM, 0, LENGTH(A.CONTRACT_NUM) - 1) = B.STOCK_CD AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_EXTERNAL_INFO AS C ON M.CUST_ID = C.CUST_ID AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_EXTERNAL_INFO AS C2 ON (CASE WHEN C.ORG_FULLNAME LIKE '%省政府' THEN REPLACE(C.ORG_FULLNAME, '省政府', '省财政厅') WHEN C.ORG_FULLNAME LIKE '%自治区政府' THEN REPLACE(C.ORG_FULLNAME, '自治区政府', '自治区财政厅') WHEN C.ORG_FULLNAME LIKE '%市人民政府' THEN REPLACE(C.ORG_FULLNAME, '市人民政府', '市财政局') WHEN C.ORG_FULLNAME LIKE '%市政府' THEN REPLACE(C.ORG_FULLNAME, '市政府', '市财政局') WHEN C.ORG_FULLNAME LIKE '%省人民政府' THEN REPLACE(C.ORG_FULLNAME, '省人民政府', '省财政厅') WHEN C.ORG_FULLNAME LIKE '%自治区人民政府' THEN REPLACE(C.ORG_FULLNAME, '自治区人民政府', '自治区财政厅') END) = C2.ORG_FULLNAME AND C2.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS D ON M.CUST_ID = D.CUST_ID AND D.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS T3 ON T3.BASIC_CCY = A.CURR_CD AND T3.FORWARD_CCY = 'CNY' AND T3.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_ACCT_FUND_INVEST AS M ON M.ACCT_NUM || '_' || M.ACCT_NO = A.CONTRACT_NUM AND M.DATA_DATE = IS_DATE AND M.GL_ITEM_CODE IN ('15030103', '11010101', '15010103', '15030101', '11010103', '11010102', '15030102', '15010101', '15010102') AND M.DATE_SOURCESD = '债券投资' AND A.ACCT_NO = M.ACCT_NO
- INNER JOIN SMTMODS.L_AGRE_BOND_INFO AS B ON M.ACCT_NUM = B.STOCK_CD AND B.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_203_ZQTZFS'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `CAST(A.TRAN_DT AS TEXT) = SUBSTRING(IS_DATE, 1, 6)`
4. `A.PRODUCT_NAME = '债券投资交易'`
5. `NOT A.ITEM_CD IN ('21010101')`
6. `NOT A.PORTFOLIO_ID IN ('1037')`
7. `A.PRODUCT_NAME IN ('债券/存单还本交易')`

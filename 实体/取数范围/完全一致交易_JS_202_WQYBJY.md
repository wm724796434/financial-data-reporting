# 完全一致交易_JS_202_WQYBJY — 取数范围

> 来源程序: `bsp_sp_js_202_wqybjy.prc`

## 数据源表（FROM）

- (SELECT D.*, ROW_NUMBER() OVER (PARTITION BY CUST_NAM ORDER BY D.SIGN_DATE DESC NULLS FIRST) AS RN FROM JS_202_WQYBJY_TMP4 AS D) AS D
- JS_202_WQYBJY_TMP5 AS b
- JS_202_WQYBJY_TMP7 AS B1
- L_PUBL_ORG_BRA_TMP AS OB
- PBOCD_DATACORE.L_CODE_DICTIONARY AS F
- PBOCD_DATACORE.L_CODE_DICTIONARY AS F1
- PBOCD_DATACORE.L_CUST_C_TMP AS B
- SMTMODS.L_CUST_P AS A

## 关联条件（JOIN）

- INNER JOIN PBOCD_DATACORE.L_CUST_C_TMP AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE AND B.CUST_TYP <> '3'
- INNER JOIN SMTMODS.L_CUST_P AS A ON T.CUST_ID = A.CUST_ID AND A.DATA_DATE = IS_DATE
- INNER JOIN JS_202_WQYBJY_TMP5 AS b ON (t.oppo_acct_num = b.acct_num)
- INNER JOIN JS_202_WQYBJY_TMP7 AS B1 ON T.OPPO_ACCT_NAM1 = B1.CUST_NAM
- INNER JOIN JS_202_WQYBJY_TMP5 AS b ON (t.oppo_acct_num = b.passbook_acct_num)
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS F ON a.ID_TYPE = F.L_CODE AND F.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS F1 ON A.OPPO_ID_TYPE = F1.L_CODE AND F1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN (SELECT D.*, ROW_NUMBER() OVER (PARTITION BY CUST_NAM ORDER BY D.SIGN_DATE DESC NULLS FIRST) AS RN FROM JS_202_WQYBJY_TMP4 AS D) AS D ON A.OPPO_ACCT_NAM = D.CUST_NAM AND D.RN = 1
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = A.NBJGH AND OB.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `TABLE_NAME = 'JS_202_WQYBJY'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `A.DATA_DATE = VS_ORDERDATE`
4. `ISWAGES = 'Y'`
5. `TRANS_CHANNEL <> 'EIBS'`
6. `JLS = 1`
7. `A.RN = 1`
8. `NOT EXISTS(SELECT 1 FROM JS_202_WQYBJY_TMP2 AS T WHERE T.KEY_TRANS_NO || T.REFERENCE_NUM = T.KEY_TRANS_NO || T.REFERENCE_NUM)`
9. `NOT A.TRAN_CODE_DESCRIBE IN ('水费', '其他', '补贴', '乙醇代发', '助学金', '奖金', '报销款', '支公积金', '网银批量', '批量代发其他', '补偿', '批量转账贷方入账失败 借方回冲', '差旅费', '占地款')`
10. `TABLE_NAME = 'PBOCD_JS_202_WQYBJY'`
11. `A.CJRQ = IS_DATE`
12. `A.OPPO_ACCT_NAM NOT LIKE '%居民委员会%'`
13. `A.OPPO_ACCT_NAM NOT LIKE '%村民委员会%'`

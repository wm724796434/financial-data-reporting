- INNER JOIN SMTMODS.L_CUST_C AS A ON A.DATA_DATE = IS_DATE AND A.CUST_ID = T.CUST_ID AND a.CUST_TYP <> '3'
- INNER JOIN SMTMODS.L_AGRE_BILL_INFO AS B ON T.ACCT_NUM = B.BILL_NUM AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.PUB_KJDK AS M2 ON SUBSTRING(M2.HYDL, 1, 3) = 'HTS' AND T.LOAN_PURPOSE_CD = M2.CODE
- INNER JOIN SMTMODS.PUB_KJDK AS M3 ON SUBSTRING(M3.HYDL, 1, 2) = 'PA' AND T.LOAN_PURPOSE_CD = M3.CODE
- INNER JOIN SMTMODS.L_PUBL_RATE AS TT ON TT.DATA_DATE = IS_DATE AND TT.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND TT.BASIC_CCY = T.CURR_CD AND TT.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.L_AGRE_LOAN_CONTRACT AS C ON C.CONTRACT_NUM = T.ACCT_NUM AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_AGRE_LOAN_CONTRACT AS C ON T.ACCT_NUM = C.CONTRACT_NUM AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS A ON A.DATA_DATE = IS_DATE AND A.CUST_ID = T.CUST_ID AND A.CUST_TYP <> '3'
- INNER JOIN SMTMODS.L_CUST_ALL AS A2 ON A2.DATA_DATE = IS_DATE AND A2.CUST_ID = T.CUST_ID
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.DATA_DATE = IS_DATE AND U.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND U.BASIC_CCY = T.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.PUB_KJDK AS M2 ON SUBSTRING(M2.HYDL, 1, 3) = 'HTS' AND (CASE WHEN COALESCE(C.HIGH_TECH_SRVE, '0') <> '0' AND T.ITEM_CD NOT LIKE '1301%' THEN C.HIGH_TECH_SRVE WHEN SUBSTRING(T.HIGH_TECH_SRVE, 1, 1) IN ('1', '2', '3', '4', '5', '6', '7', '8') AND T.ITEM_CD LIKE '1301%' THEN SUBSTRING(T.HIGH_TECH_SRVE, 1, 1) END) = M2.CODE
- INNER JOIN SMTMODS.PUB_KJDK AS M3 ON SUBSTRING(M3.HYDL, 1, 2) = 'PA' AND (CASE WHEN COALESCE(C.PANT_DENS_INDU, '0') <> '0' AND T.ITEM_CD NOT LIKE '1301%' THEN C.PANT_DENS_INDU WHEN SUBSTRING(T.PANT_DENS_INDU, 1, 2) IN ('01', '02', '03', '04', '05', '06', '07') AND T.ITEM_CD LIKE '1301%' THEN SUBSTRING(T.PANT_DENS_INDU, 1, 2) END) = M3.CODE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE

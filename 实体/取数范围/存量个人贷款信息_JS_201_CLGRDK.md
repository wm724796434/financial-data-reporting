# 存量个人贷款信息_JS_201_CLGRDK — 取数范围

> 来源程序: `bsp_sp_js_201_clgrdk.prc`

## 数据源表（FROM）

- (SELECT ROW_NUMBER() OVER (PARTITION BY T.CUST_ID ORDER BY T.CUST_ID DESC NULLS FIRST) AS RN, T.* FROM JS_102_GRKHXX AS T WHERE T.DATA_DATE = IS_DATE) AS C
- DBFS_TMP AS TP7
- L_ACCT_LOAN_SUOQI AS T1
- L_PUBL_ORG_BRA_TMP AS OB
- PBOCD_JS_102_GRKHXX AS AA
- PBOCD_JS_201_CLGRDK_SQ AS BK
- SMTMODS.L_PUBL_RATE AS R
- SMTMODS.v_pub_idx_dk_zqdqrjj AS ZQ

## 关联条件（JOIN）

- INNER JOIN (SELECT ROW_NUMBER() OVER (PARTITION BY T.CUST_ID ORDER BY T.CUST_ID DESC NULLS FIRST) AS RN, T.* FROM JS_102_GRKHXX AS T WHERE T.DATA_DATE = IS_DATE) AS C ON T.CUST_ID = C.CUST_ID AND C.CJRQ = IS_DATE AND C.RN = 1
- INNER JOIN SMTMODS.L_PUBL_RATE AS R ON R.DATA_DATE = IS_DATE AND R.BASIC_CCY = T.CURR_CD AND R.FORWARD_CCY = 'CNY' AND R.DATA_DATE = IS_DATE
- INNER JOIN DBFS_TMP AS TP7 ON T.LOAN_NUM = TP7.LOAN_NUM
- INNER JOIN SMTMODS.v_pub_idx_dk_zqdqrjj AS ZQ ON T.LOAN_NUM = ZQ.LOAN_NUM AND T.DATA_DATE = ZQ.DATA_DATE
- INNER JOIN L_ACCT_LOAN_SUOQI AS T1 ON T.LOAN_NUM = T1.LOAN_NUM AND T.MATURITY_DT_BEFORE = T1.MATURITY_DT_BEFORE AND T.MATURITY_DT = T1.MATURITY_DT
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_JS_201_CLGRDK_SQ AS BK ON T.LOAN_NUM = BK.LOAN_NUM AND BK.CJRQ = VS_LAST_TEXT
- INNER JOIN PBOCD_JS_102_GRKHXX AS AA ON T.CUST_ID_NO = AA.CUST_ID_NO AND AA.CJRQ = IS_DATE AND AA.FRNBJGH = '990000'

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_201_CLGRDK_TMP'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `T.DATA_DATE = IS_DATE`
4. `T.LOAN_ACCT_BAL > 0`
5. `T.CURR_CD = 'CNY'`
6. `T.ACCT_TYP LIKE '01%'`
7. `T.CANCEL_FLG = 'N'`
8. `T.LOAN_STOCKEN_DATE IS NULL`
9. `TRIM(T.DATA_DATE) = VS_TEXT`
10. `CJRQ = IS_DATE`

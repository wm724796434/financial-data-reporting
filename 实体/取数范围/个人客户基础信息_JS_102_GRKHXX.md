## 数据源表（FROM）

- (SELECT B.CUST_ID, B.DRAWDOWN_DT, B.DEPARTMENTD, ROW_NUMBER() OVER (PARTITION BY B.CUST_ID ORDER BY B.DRAWDOWN_DT DESC NULLS FIRST) AS RN FROM JS_102_GRKHXX_TEMP10 AS B) AS B
- (SELECT DISTINCT ID_CARD FROM TMP_JS_102_GRKHXX_GLF) AS GLF
- (SELECT T.CUST_ID, SUM(COALESCE(T.FACILITY_AMT, 0)) AS FACILITY_AMT, SUM(COALESCE(T.USED_FACILITY_AMT, 0)) AS USED_FACILITY_AMT FROM SMTMODS.L_AGRE_CREDITLINE AS T WHERE T.DATA_DATE = IS_DATE AND T.FACILITY_STS = 'Y' GROUP BY T.CUST_ID) AS T2
- JS_102_GRKHXX_TEMP01 AS T1
- JS_102_GRKHXX_TEMP05 AS T5
- JS_102_GRKHXX_TEMP09 AS T9
- L_CODE_DICTIONARY AS D1
- L_CODE_DICTIONARY AS D2
- L_CODE_DICTIONARY AS D3
- L_CODE_DICTIONARY AS D4
- L_CODE_DICTIONARY AS D5
- L_CODE_DICTIONARY AS D6
- L_CUST_P_NEW AS B

- PBOCD_JS_102_GRKHXX_SQ AS BK
- SMTMODS.L_ACCT_LOAN AS B
- SMTMODS.L_ACCT_WRITE_OFF AS XX
- SMTMODS.L_CUST_ALL AS A

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_WRITE_OFF AS XX ON T.LOAN_NUM = XX.LOAN_NUM AND XX.DATA_DATE = IS_DATE AND SUBSTRING(CAST(XX.WRITE_OFF_DATE AS TEXT), 1, 6) = SUBSTRING(IS_DATE, 1, 6)
- INNER JOIN L_CUST_P_NEW AS B ON A.CUST_ID = B.CUST_ID
- INNER JOIN SMTMODS.L_ACCT_LOAN AS B ON A.LOAN_NUM = B.LOAN_NUM AND B.DATA_DATE = is_date
- INNER JOIN L_CUST_P_NEW AS C ON B.CUST_ID = C.CUST_ID
- INNER JOIN JS_102_GRKHXX_TEMP05 AS T5 ON T.CUST_ID = T5.CUST_ID
- INNER JOIN SMTMODS.L_CUST_ALL AS A ON T.CUST_ID = A.CUST_ID AND A.DATA_DATE = IS_DATE
- INNER JOIN (SELECT T.CUST_ID, SUM(COALESCE(T.FACILITY_AMT, 0)) AS FACILITY_AMT, SUM(COALESCE(T.USED_FACILITY_AMT, 0)) AS USED_FACILITY_AMT FROM SMTMODS.L_AGRE_CREDITLINE AS T WHERE T.DATA_DATE = IS_DATE AND T.FACILITY_STS = 'Y' GROUP BY T.CUST_ID) AS T2 ON T.CUST_ID = T2.CUST_ID
- INNER JOIN JS_102_GRKHXX_TEMP01 AS T1 ON T.CUST_ID = T1.CUSTID
- INNER JOIN L_CODE_DICTIONARY AS D1 ON COALESCE(T.LEGAL_CARD_TYPE, T.ID_TYPE) = D1.L_CODE AND D1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS D2 ON T.NATION_CD = D2.L_CODE AND D2.CODE_CLMN_NAME = 'NATION_CD'
- INNER JOIN L_CODE_DICTIONARY AS D3 ON T.MARRIAGE_TYP = D3.L_CODE AND D3.CODE_CLMN_NAME = 'MARRIAGE_TYP'
- INNER JOIN L_CODE_DICTIONARY AS D4 ON T.CUST_NATION = D4.L_CODE AND D4.CODE_CLMN_NAME = 'CUST_NATION'
- INNER JOIN L_CODE_DICTIONARY AS D5 ON TRIM(T.EDUCATION_CD) = D5.L_CODE AND D5.CODE_CLMN_NAME = 'EDUCATION_CD'
- INNER JOIN L_CODE_DICTIONARY AS D6 ON T.SEX_TYP = D6.L_CODE AND D6.CODE_CLMN_NAME = 'SEX_TYP'
- INNER JOIN (SELECT DISTINCT ID_CARD FROM TMP_JS_102_GRKHXX_GLF) AS GLF ON T.ID_NO = GLF.ID_CARD
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB1 ON OB1.ORG_NUM = T.ORG_NUM AND OB1.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_ACCT_LOAN AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE AND B.DEPARTMENTD IN ('德惠长银', '普惠金融', '个人信贷', '磐石村镇')
- INNER JOIN (SELECT B.CUST_ID, B.DRAWDOWN_DT, B.DEPARTMENTD, ROW_NUMBER() OVER (PARTITION BY B.CUST_ID ORDER BY B.DRAWDOWN_DT DESC NULLS FIRST) AS RN FROM JS_102_GRKHXX_TEMP10 AS B) AS B ON A.CUST_ID = B.CUST_ID AND A.DRAWDOWN_DT = B.DRAWDOWN_DT AND B.RN = 1
- INNER JOIN JS_102_GRKHXX_TEMP09 AS T9 ON T.CUST_ID = T9.CUST_ID
- INNER JOIN PBOCD_JS_102_GRKHXX_SQ AS BK ON T.CUST_ID_NO = BK.CUST_ID_NO AND T.FRNBJGH = BK.FRNBJGH AND BK.CJRQ = VS_LAST_TEXT

## 筛选条件（WHERE）

1. `T.DATA_DATE = IS_DATE`
2. `B.DATA_DATE = IS_DATE`
3. `B.OPERATE_CUST_TYPE IN ('A', 'B')`
4. `T.CANCEL_FLG = 'Y'`
5. `XX.LOAN_NUM IS NULL`
6. `NOT T.CUST_ID IN (SELECT T1.CUST_ID FROM SMTMODS.L_ACCT_LOAN AS T1 WHERE T1.DATA_DATE = IS_DATE AND T1.LOAN_STOCKEN_DATE IS NULL AND T1.CANCEL_FLG = 'N')`
7. `NOT EXISTS(SELECT 1 FROM A AS A WHERE A.CUST_ID = B.CUST_ID)`
8. `TABLE_NAME = 'JS_102_GRKHXX'`
9. `PARTITION_NAME = 'P' || IS_DATE`
10. `NOT EXISTS(SELECT 1 FROM JS_102_GRKHXX_TEMP03 AS TEMP3 WHERE T.CUST_ID = TEMP3.CUST_ID)`
11. `TRIM(A.DATA_DATE) = IS_DATE`
12. `DATA_DATE = IS_DATE`
13. `DEPARTMENTD IN ('德惠长银', '普惠金融', '个人信贷', '磐石村镇')`
14. `T.CJRQ = IS_DATE`
15. `T.DATA_DATE = VS_TEXT`
16. `T.FRNBJGH = '510000'`
17. `T.CUST_ID_NO IN ('230702198304130729', '220602198702103315', '220223197609266027', '220284199206031567', '220223198004146617')`
18. `NOT EXISTS(SELECT * FROM PBOCD_JS_102_GRKHXX_TMP AS B WHERE B.CJRQ = IS_DATE AND B.FRNBJGH = '990000' AND B.CUST_ID_NO = T.CUST_ID_NO)`
19. `RN = 1`
20. `NOT EXISTS(SELECT 1 FROM PBOCD_JS_102_GRKHXX AS A WHERE A.CJRQ = IS_DATE AND A.FRNBJGH = '990000' AND T.CUST_ID_NO = A.CUST_ID_NO)`

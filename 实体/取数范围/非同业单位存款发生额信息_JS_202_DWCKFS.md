# 非同业单位存款发生额信息_JS_202_DWCKFS — 取数范围

> 来源程序: `bsp_sp_js_202_dwckfs.prc`

## 数据源表（FROM）

- (SELECT SUM(CASE WHEN GL.ITEM_CD = '3020' THEN -GL.DEBIT_BAL ELSE GL.CREDIT_BAL END) AS BALANCE FROM SMTMODS.L_FINA_GL AS GL WHERE GL.DATA_DATE = IS_DATE AND GL.ITEM_CD IN ('3010', '3020') AND GL.CURR_CD = 'CNY' AND GL.ORG_NUM = '990000') AS T1
- L_CODE_DICTIONARY AS COD
- L_CODE_DICTIONARY AS COD2
- L_PUBL_ORG_BRA_TMP AS OB
- L_PUBL_ORG_BRA_TMP AS OB1
- PBOCD_DATACORE.L_CODE_DICTIONARY AS D1
- PBOCD_DATACORE.L_CODE_DICTIONARY AS D2
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E1
- PBOCD_JS_202_FTYDWC_SQ AS T2
- SMTMODS.L_ACCT_DEPOSIT AS B
- SMTMODS.L_ACCT_DEPOSIT AS C
- SMTMODS.L_CUST_C AS A
- SMTMODS.L_PUBL_RATE AS E
- deposit_tmp01 AS aa

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_DEPOSIT AS B ON T.ACCOUNT_CODE = B.ACCT_NUM AND B.DATA_DATE = IS_DATE AND B.ACCT_TYPE <> '0602' OR B.ACCT_TYPE IS NULL
- INNER JOIN SMTMODS.L_CUST_C AS A ON T.CUST_ID = A.CUST_ID AND A.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS E ON T.CURRENCY = E.BASIC_CCY AND E.CCY_DATE = STR_TO_DATE(IS_DATE, 'yyyymmdd') AND E.DATA_DATE = IS_DATE AND E.FORWARD_CCY = 'CNY'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS D1 ON A.ID_TYPE = D1.L_CODE AND D1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS D2 ON A.NATION_CD = D2.L_CODE AND D2.CODE_CLMN_NAME = 'REG_REGION_CODE'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON B.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E1 ON B.PBOC_ACCT_NATURE_CD = E1.L_CODE AND E1.CODE_CLMN_NAME = 'ACCT_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS COD ON COD.L_CODE = T.COUNTPTY_ID_TYPE AND COD.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS COD2 ON COD2.L_CODE = T.COUNTPTY_ID_TYPE2 AND COD2.CODE_CLMN_NAME = 'COUNTPTY_ID_TYPE'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = TRIM(B.ORG_NUM) AND OB.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB1 ON OB1.ORG_NUM = CASE WHEN T.CUST_ID LIKE '2999%' THEN SUBSTRING(T.CUST_ID, 5) ELSE '#' END AND OB1.DATA_DATE = IS_DATE
- INNER JOIN (SELECT SUM(CASE WHEN GL.ITEM_CD = '3020' THEN -GL.DEBIT_BAL ELSE GL.CREDIT_BAL END) AS BALANCE FROM SMTMODS.L_FINA_GL AS GL WHERE GL.DATA_DATE = IS_DATE AND GL.ITEM_CD IN ('3010', '3020') AND GL.CURR_CD = 'CNY' AND GL.ORG_NUM = '990000') AS T1 ON 1 = 1
- INNER JOIN PBOCD_JS_202_FTYDWC_SQ AS T2 ON T.O_ACCT_NUM = T2.DEP_ACC_CODE AND T2.CJRQ = VS_LAST_TEXT AND T2.PRODUCT_TYPE = 'D15'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = TRIM(T.ORG_NUM) AND OB.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON T.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E1 ON T.PBOC_ACCT_NATURE_CD = E1.L_CODE AND E1.CODE_CLMN_NAME = 'ACCT_TYPE'
- INNER JOIN SMTMODS.L_ACCT_DEPOSIT AS B ON A.DEP_ACC_CODE = B.ACCT_NUM AND B.DATA_DATE = VS_LAST_TEXT
- INNER JOIN SMTMODS.L_ACCT_DEPOSIT AS C ON A.DEP_ACC_CODE = C.ACCT_NUM AND C.DATA_DATE = IS_DATE
- INNER JOIN deposit_tmp01 AS aa ON t.dep_acc_code = aa.dep_acc_code

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_202_DWCKFS_TMP'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `TABLE_NAME = 'PBOCD_JS_202_DWCKFS'`
4. `T.GL_ITEM_CODE IN ('20110202', '20110203', '20110204', '20110207', '20110205', '20110209', '20110210', '20110201', '20110208', '20110211') OR T.GL_ITEM_CODE IS NULL OR SUBSTRING(T.GL_ITEM_CODE, 1, 6) IN ('201103', '201104', '201105', '201106') OR SUBSTRING(T.GL_ITEM_CODE, 1, 4) IN ('2008', '2009')`
5. `T.DATA_DATE BETWEEN VS_ORDERDATE AND IS_DATE`
6. `T.TRANS_AMT <> 0`
7. `NOT T.TRAN_CODE_DESCRIBE IN ('转久悬', '久悬激活') OR (T.TRAN_CODE_DESCRIBE = '久悬激活' AND TRANTYPE2_DESC = '营业外激活')`
8. `NOT T.ACCOUNT_CODE IS NULL`
9. `(B.GL_ITEM_CODE IN ('20110202', '20110203', '20110204', '20110211', '20110207', '20110205', '20110201', '20110208', '20110209', '20110210', '22410101') OR SUBSTRING(B.GL_ITEM_CODE, 1, 6) IN ('201103', '201104', '201105', '201106') OR SUBSTRING(B.GL_ITEM_CODE, 1, 4) IN ('2008', '2009')) AND NOT A.DEPOSIT_CUSTTYPE IN ('13', '14')`
10. `ROUND(T.TRANS_AMT * E.CCY_RATE, 2) <> 0`
11. `T.O_ACCT_NUM = '7330140601000020_1'`
12. `T.DATA_DATE = IS_DATE`
13. `ABS(COALESCE(T1.BALANCE, 0) - COALESCE(T2.BALANCE, 0)) <> 0`
14. `A.CJRQ = VS_LAST_TEXT`
15. `B.GL_ITEM_CODE <> '63010401'`
16. `C.GL_ITEM_CODE = '63010401'`
17. `A.CJRQ = IS_DATE`
18. `B.GL_ITEM_CODE = '63010401'`
19. `C.GL_ITEM_CODE <> '63010401'`
20. `C.ACCT_STS = 'C'`
21. `t.CJRQ = VS_LAST_TEXT`

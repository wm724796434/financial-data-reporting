# 同业存款发生额信息_JS_202_TYCKFS — 取数范围

> 来源程序: `bsp_sp_js_202_tyckfs.prc`

## 数据源表（FROM）

- (SELECT * FROM (SELECT ECIF_CUST_ID, FINA_CODE_NEW, CUST_BANK_CD, LEGAL_FLAG, ROW_NUMBER() OVER (PARTITION BY ECIF_CUST_ID ORDER BY LEGAL_FLAG DESC NULLS FIRST, FINA_CODE_NEW NULLS LAST, CUST_BANK_CD NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS TY WHERE DATA_DATE = IS_DATE) AS TY WHERE TY.RN = 1) AS TY
- (SELECT D.ACCT_NUM, D.ACCT_TYP, REAL_INT_RAT, D.DEPOSIT_TERM_FLG, D.ACCT_STS, D.ACCT_ATTR, MAX(D.MATURE_DATE) OVER (PARTITION BY D.ACCT_NUM) AS MATURE_DATE, MIN(D.START_DATE) OVER (PARTITION BY D.ACCT_NUM) AS START_DATE, ROW_NUMBER() OVER (PARTITION BY D.ACCT_NUM ORDER BY D.MATURE_DATE DESC NULLS FIRST) AS RN FROM smtmods.l_acct_fund_mmfund AS d WHERE d.data_date = IS_DATE AND d.deposit_term_flg LIKE 'A%' AND SUBSTRING(D.acct_typ, 1, 3) IN ('201', '101')) AS D
- (SELECT DISTINCT B.DEP_ACC_CODE, B.DEP_AGR_CODE, B.CONT_PARTY_CODE, B.CUST_NAME, B.CONT_PARTY_TYPE FROM PBOCD_JS_202_CLTYCK AS B WHERE B.CJRQ = IS_DATE AND B.PRODUCT_TYPE <> 'T03') AS BK2
- (SELECT DISTINCT B.DEP_ACC_CODE, B.DEP_AGR_CODE, B.CONT_PARTY_CODE, B.CUST_NAME, B.CONT_PARTY_TYPE FROM PBOCD_JS_202_CLTYCK_SQ AS B WHERE B.CJRQ = VS_LAST_TEXT AND B.PRODUCT_TYPE <> 'T03') AS BK1
- (SELECT DISTINCT T.CUST_NAME, T.CUST_ID_NO, T.CUST_ID_NO_SOURCE, T.CONT_PARTY_TYPE FROM JS_102_TYKHXX_CODE AS T WHERE T.CUST_ID_NO_SOURCE <> '2999999999') AS CD
- (SELECT DISTINCT T.CUST_NAME, T.CUST_ID_NO, T.CUST_NAME_SOURCE, T.CONT_PARTY_TYPE FROM JS_102_TYKHXX_CODE AS T) AS CD1
- L_CUST_C_TMP AS T1
- L_PUBL_ORG_BRA_TMP AS OB
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E2
- PBOCD_JS_101_JRJGFZ AS T2
- SMTMODS.L_ACCT_DEPOSIT AS AD
- SMTMODS.L_ACCT_FUND_CDS_BAL AS T
- SMTMODS.L_CUST_ALL AS B
- SMTMODS.L_CUST_C AS T1
- SMTMODS.L_PUBL_RATE AS T3
- smtmods.l_acct_fund_mmfund AS D

## 关联条件（JOIN）

- INNER JOIN smtmods.l_acct_fund_mmfund AS D ON a.account_code = D.acct_num AND D.data_date = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS T3 ON T3.BASIC_CCY = A.CURRENCY AND T3.FORWARD_CCY = 'CNY' AND T3.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON A.ORG_NUM = OB.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN L_CUST_C_TMP AS T1 ON A.CUST_ID = T1.CUST_ID AND T1.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_ACCT_DEPOSIT AS AD ON A.CUST_ID = AD.CUST_ID AND A.ACCOUNT_CODE = AD.ACCT_NUM AND AD.DATA_DATE = IS_DATE
- INNER JOIN (SELECT * FROM (SELECT ECIF_CUST_ID, FINA_CODE_NEW, CUST_BANK_CD, LEGAL_FLAG, ROW_NUMBER() OVER (PARTITION BY ECIF_CUST_ID ORDER BY LEGAL_FLAG DESC NULLS FIRST, FINA_CODE_NEW NULLS LAST, CUST_BANK_CD NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS TY WHERE DATA_DATE = IS_DATE) AS TY WHERE TY.RN = 1) AS TY ON D.CUST_ID = TY.ECIF_CUST_ID
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON D.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E2 ON TRIM(TY.FINA_CODE_NEW) = E2.L_CODE AND E2.CODE_CLMN_NAME = 'FINA_CODE_NEW'
- INNER JOIN (SELECT D.ACCT_NUM, D.ACCT_TYP, REAL_INT_RAT, D.DEPOSIT_TERM_FLG, D.ACCT_STS, D.ACCT_ATTR, MAX(D.MATURE_DATE) OVER (PARTITION BY D.ACCT_NUM) AS MATURE_DATE, MIN(D.START_DATE) OVER (PARTITION BY D.ACCT_NUM) AS START_DATE, ROW_NUMBER() OVER (PARTITION BY D.ACCT_NUM ORDER BY D.MATURE_DATE DESC NULLS FIRST) AS RN FROM smtmods.l_acct_fund_mmfund AS d WHERE d.data_date = IS_DATE AND d.deposit_term_flg LIKE 'A%' AND SUBSTRING(D.acct_typ, 1, 3) IN ('201', '101')) AS D ON a.account_code = D.ACCT_NUM AND RN = 1
- INNER JOIN (SELECT * FROM (SELECT ECIF_CUST_ID, FINA_CODE_NEW, CUST_BANK_CD, LEGAL_FLAG, ROW_NUMBER() OVER (PARTITION BY ECIF_CUST_ID ORDER BY LEGAL_FLAG DESC NULLS FIRST, FINA_CODE_NEW NULLS LAST, CUST_BANK_CD NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS TY WHERE DATA_DATE = IS_DATE) AS TY WHERE TY.RN = 1) AS TY ON A.CUST_ID = TY.ECIF_CUST_ID
- INNER JOIN SMTMODS.L_CUST_C AS T1 ON T.CUST_ID = T1.CUST_ID AND T1.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_JS_101_JRJGFZ AS T2 ON T.ORG_NUM = T2.ORG_NUM AND T2.CJRQ = VS_MAX_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = TRIM(T.ORG_NUM) AND OB.DATA_DATE = IS_DATE
- INNER JOIN (SELECT DISTINCT B.DEP_ACC_CODE, B.DEP_AGR_CODE, B.CONT_PARTY_CODE, B.CUST_NAME, B.CONT_PARTY_TYPE FROM PBOCD_JS_202_CLTYCK_SQ AS B WHERE B.CJRQ = VS_LAST_TEXT AND B.PRODUCT_TYPE <> 'T03') AS BK1 ON T.DEP_ACC_CODE = BK1.DEP_ACC_CODE
- INNER JOIN (SELECT DISTINCT B.DEP_ACC_CODE, B.DEP_AGR_CODE, B.CONT_PARTY_CODE, B.CUST_NAME, B.CONT_PARTY_TYPE FROM PBOCD_JS_202_CLTYCK AS B WHERE B.CJRQ = IS_DATE AND B.PRODUCT_TYPE <> 'T03') AS BK2 ON T.DEP_ACC_CODE = BK2.DEP_ACC_CODE
- INNER JOIN (SELECT DISTINCT T.CUST_NAME, T.CUST_ID_NO, T.CUST_ID_NO_SOURCE, T.CONT_PARTY_TYPE FROM JS_102_TYKHXX_CODE AS T WHERE T.CUST_ID_NO_SOURCE <> '2999999999') AS CD ON COALESCE(COALESCE(T1.TYSHXYDM, T1.ID_NO), T1.ID_NO) = CD.CUST_ID_NO_SOURCE
- INNER JOIN (SELECT DISTINCT T.CUST_NAME, T.CUST_ID_NO, T.CUST_NAME_SOURCE, T.CONT_PARTY_TYPE FROM JS_102_TYKHXX_CODE AS T) AS CD1 ON COALESCE(T.CUST_NAME, T1.CUST_NAM) = CD1.CUST_NAME_SOURCE
- INNER JOIN SMTMODS.L_ACCT_FUND_CDS_BAL AS T ON A.CONTRACT_NUM = T.ACCT_NUM AND T.DATA_DATE = IS_DATE AND SUBSTRING(GL_ITEM_CODE, 1, 6) IN ('250202') OR GL_ITEM_CODE IN ('11010105', '15030105')
- INNER JOIN SMTMODS.L_PUBL_RATE AS T3 ON T3.BASIC_CCY = T.CURR_CD AND T3.FORWARD_CCY = 'CNY' AND T3.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = TRIM(A.ORG_NUM) AND OB.DATA_DATE = IS_DATE
- INNER JOIN (SELECT * FROM (SELECT ECIF_CUST_ID, FINA_CODE_NEW, CUST_BANK_CD, LEGAL_FLAG, ROW_NUMBER() OVER (PARTITION BY ECIF_CUST_ID ORDER BY LEGAL_FLAG DESC NULLS FIRST, FINA_CODE_NEW NULLS LAST, CUST_BANK_CD NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS TY WHERE DATA_DATE = IS_DATE) AS TY WHERE TY.RN = 1) AS TY ON T.CUST_ID = TY.ECIF_CUST_ID
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON T.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN SMTMODS.L_ACCT_FUND_CDS_BAL AS T ON A.CONTRACT_NUM = T.CDS_NO AND A.CONT_PARTY_NAME = T.CONT_PARTY_NAME AND T.DATA_DATE = IS_DATE AND SUBSTRING(GL_ITEM_CODE, 1, 6) IN ('250202') OR GL_ITEM_CODE IN ('11010105', '15030105')

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_202_TYCKFS_TMP'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `A.DATA_DATE BETWEEN SUBSTRING(IS_DATE, 1, 6) || '01' AND IS_DATE`
4. `A.TRANS_AMT <> 0`
5. `SUBSTRING(GL_ITEM_CODE, 1, 4) IN ('1011', '1031', '2012')`
6. `SUBSTRING(D.acct_typ, 1, 3) IN ('201', '101')`
7. `D.DEPOSIT_TERM_FLG LIKE 'B%'`
8. `d.data_date = IS_DATE`
9. `d.deposit_term_flg LIKE 'A%'`
10. `PRODUCT_TYPE IN ('T012', 'T022')`
11. `PRODUCT_TYPE IN ('T011', 'T021')`
12. `TABLE_NAME = 'PBOCD_JS_202_TYCKFS'`
13. `T.CJRQ = IS_DATE`
14. `CAST(A.TRAN_DT AS TEXT) = SUBSTRING(IS_DATE, 1, 6)`
15. `T.DATE_SOURCESD = '存单投资'`
16. `T.DATE_SOURCESD = '存单发行'`

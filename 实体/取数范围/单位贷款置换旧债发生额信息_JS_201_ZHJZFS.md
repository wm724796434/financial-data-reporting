# 单位贷款置换旧债发生额信息_JS_201_ZHJZFS — 取数范围

> 来源程序: `bsp_sp_js_201_zhjzfs.prc`

## 数据源表（FROM）

- PBOCD_DATACORE.L_CODE_DICTIONARY AS D1
- SMTMODS.L_ACCT_LOAN AS A1
- SMTMODS.L_CUST_C AS B
- SMTMODS.L_CUST_C AS B2
- SMTMODS.L_CUST_P AS B3
- SMTMODS.L_PUBL_ORG_BRA AS C

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_CUST_C AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE AND B.CUST_TYP <> '3'
- INNER JOIN SMTMODS.L_PUBL_ORG_BRA AS C ON A.ORG_NUM = C.ORG_NUM AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_ACCT_LOAN AS A1 ON A.LOAN_NUM_OLD = A1.LOAN_NUM AND A1.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS B2 ON A1.CUST_ID = B2.CUST_ID AND B2.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS B3 ON A1.CUST_ID = B3.CUST_ID AND B3.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS D1 ON B3.ID_TYPE = D1.L_CODE AND D1.CODE_CLMN_NAME = 'ID_TYPE'

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_201_ZHJZFS'`
2. `PARTITION_NAME = 'PBOCD_JS_201_ZHJZFS_' || IS_DATE`
3. `A.DATA_DATE = IS_DATE`
4. `A.CANCEL_FLG = 'N'`
5. `A.REPAY_FLG = 'Y'`
6. `A.LOAN_ACCT_BAL <> 0`
7. `DATE_TRUNC('MM', A.DRAWDOWN_DT) = DATE_TRUNC('MM', STR_TO_DATE(IS_DATE, '%Y%m%d'))`
8. `A.LOAN_STOCKEN_DATE IS NULL`

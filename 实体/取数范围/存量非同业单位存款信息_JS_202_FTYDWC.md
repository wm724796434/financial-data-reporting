# 存量非同业单位存款信息_JS_202_FTYDWC — 取数范围

> 来源程序: `bsp_sp_js_202_ftydwc.prc`

## 数据源表（FROM）

- (SELECT SUM(CASE WHEN GL.ITEM_CD IN ('3020', '304002') THEN -GL.DEBIT_BAL ELSE GL.CREDIT_BAL END) AS BALANCE FROM SMTMODS.L_FINA_GL AS GL WHERE GL.DATA_DATE = IS_DATE AND GL.ITEM_CD IN ('3010', '3020', '303002', '304002') AND GL.CURR_CD = 'CNY' AND GL.ORG_NUM = '990000') AS T1
- L_PUBL_ORG_BRA_TMP AS OB
- L_PUBL_ORG_BRA_TMP AS OB1
- PBOCD_DATACORE.L_CODE_DICTIONARY AS D1
- PBOCD_DATACORE.L_CODE_DICTIONARY AS D2
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E1
- SMTMODS.L_CUST_C AS C
- SMTMODS.L_PUBL_RATE AS B

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_PUBL_RATE AS B ON A.CURR_CD = B.BASIC_CCY AND B.CCY_DATE = STR_TO_DATE(IS_DATE, 'yyyymmdd') AND A.DATA_DATE = B.DATA_DATE AND B.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.L_CUST_C AS C ON A.CUST_ID = C.CUST_ID AND C.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS D1 ON C.ID_TYPE = D1.L_CODE AND D1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS D2 ON C.NATION_CD = D2.L_CODE AND D2.CODE_CLMN_NAME = 'REG_REGION_CODE'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON A.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E1 ON A.PBOC_ACCT_NATURE_CD = E1.L_CODE AND E1.CODE_CLMN_NAME = 'ACCT_TYPE'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = A.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB1 ON OB1.ORG_NUM = CASE WHEN A.CUST_ID LIKE '2999%' THEN SUBSTRING(A.CUST_ID, 5) ELSE '#' END AND OB1.DATA_DATE = IS_DATE
- INNER JOIN (SELECT SUM(CASE WHEN GL.ITEM_CD IN ('3020', '304002') THEN -GL.DEBIT_BAL ELSE GL.CREDIT_BAL END) AS BALANCE FROM SMTMODS.L_FINA_GL AS GL WHERE GL.DATA_DATE = IS_DATE AND GL.ITEM_CD IN ('3010', '3020', '303002', '304002') AND GL.CURR_CD = 'CNY' AND GL.ORG_NUM = '990000') AS T1 ON 1 = 1
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON T.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E1 ON T.PBOC_ACCT_NATURE_CD = E1.L_CODE AND E1.CODE_CLMN_NAME = 'ACCT_TYPE'

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_202_FTYDWC_TMP'`
2. `PARTITION_NAME = 'PBOCD_JS_202_FTYDWC_TMP_' || IS_DATE`
3. `TABLE_NAME = 'PBOCD_JS_202_FTYDWC'`
4. `PARTITION_NAME = 'PBOCD_JS_202_FTYDWC_' || IS_DATE`
5. `(A.GL_ITEM_CODE IN ('20110202', '20110203', '20110204', '20110211', '20110207', '20110205', '20110201', '20110208', '20110209', '20110210', '22410101') OR SUBSTRING(A.GL_ITEM_CODE, 1, 6) IN ('201103', '201104', '201105', '201106') OR SUBSTRING(A.GL_ITEM_CODE, 1, 4) IN ('2008', '2009')) AND NOT C.DEPOSIT_CUSTTYPE IN ('13', '14')`
6. `A.DATA_DATE = IS_DATE`
7. `A.ACCT_BALANCE > 0`
8. `ROUND(A.ACCT_BALANCE * B.CCY_RATE, 2) > 0`
9. `T.O_ACCT_NUM = '7330140601000020_1'`
10. `T.DATA_DATE = IS_DATE`
11. `A.CJRQ = IS_DATE`
12. `A.FRNBJGH = '990000' OR (A.FRNBJGH >= '520000' AND A.FRNBJGH <= '600000')`
13. `NOT EXISTS(SELECT 1 FROM PBOCD_JS_202_FTYDWC AS B WHERE B.CJRQ = IS_DATE AND A.DEP_ACC_CODE = B.DEP_ACC_CODE)`

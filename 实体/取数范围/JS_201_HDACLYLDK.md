# JS_201_HDACLYLDK — 取数范围

> 来源程序: `bsp_sp_js_201_hdaclyldk.prc`

## 数据源表（FROM）

- L_PUBL_ORG_BRA_TMP AS OB
- SMTMODS.L_ACCT_LOAN AS T1
- SMTMODS.L_AGRE_LOAN_CONTRACT AS D
- SMTMODS.L_CUST_ALL AS A2
- SMTMODS.L_CUST_C AS C
- SMTMODS.L_PUBL_RATE AS U
- SMTMODS.QGYL_LIST AS F

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_LOAN AS T1 ON T1.DATA_DATE = IS_DATE AND T.ACCT_NUM || T.DRAFT_RNG = T1.ACCT_NUM || T1.DRAFT_RNG AND SUBSTRING(T1.ITEM_CD, 1, 6) IN ('130101', '130104') AND SUBSTRING(T.ITEM_CD, 1, 6) IN ('130102', '130105')
- INNER JOIN SMTMODS.L_AGRE_LOAN_CONTRACT AS D ON D.DATA_DATE = IS_DATE AND T.ACCT_NUM = D.CONTRACT_NUM AND SUBSTRING(D.PENSION_INDUSTRY, 1, 2) IN ('01', '02', '03', '04', '05', '06', '07', '08', '09', '10', '11', '12')
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.DATA_DATE = IS_DATE AND U.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND U.BASIC_CCY = T.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.QGYL_LIST AS F ON T.LOAN_NUM = F.LOAN_NUM
- INNER JOIN SMTMODS.L_CUST_ALL AS A2 ON A2.DATA_DATE = IS_DATE AND A2.CUST_ID = T.CUST_ID
- INNER JOIN SMTMODS.L_CUST_C AS C ON C.DATA_DATE = IS_DATE AND C.CUST_ID = T.CUST_ID
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `T.DATA_DATE = IS_DATE`
2. `T.LOAN_ACCT_BAL > 0`
3. `T.ACCT_STS <> '3'`
4. `T.CANCEL_FLG <> 'Y'`
5. `T.LOAN_STOCKEN_DATE IS NULL`
6. `T.CANCEL_FLG = 'N'`
7. `(NOT D.PENSION_INDUSTRY IS NULL AND (SUBSTRING(T.ITEM_CD, 1, 4) IN ('1303', '1305', '1306') OR (T.CURR_CD = 'CNY' AND T.ACCT_TYP LIKE '01%'))) OR (COALESCE(T.LOAN_ACCT_BAL, 0) + COALESCE(T.INT_ADJEST_AMT, 0) > 0 AND NOT T.PENSION_INDUSTRY IS NULL AND SUBSTRING(T.ITEM_CD, 1, 6) IN ('130101', '130104'))`
8. `SUBSTRING(T.ITEM_CD, 1, 4) IN ('1303', '1305', '1306') OR (T.CURR_CD = 'CNY' AND T.ACCT_TYP LIKE '01%') OR (COALESCE(T.LOAN_ACCT_BAL, 0) + COALESCE(T.INT_ADJEST_AMT, 0) > 0 AND SUBSTRING(T.ITEM_CD, 1, 6) IN ('130101', '130104'))`
9. `EXISTS(SELECT 1 FROM SMTMODS.QGYL_LIST AS X WHERE X.LOAN_NUM = T.LOAN_NUM)`
10. `NOT F.LOAN_NUM IS NULL OR (NOT D.PENSION_INDUSTRY IS NULL AND (SUBSTRING(T.ITEM_CD, 1, 4) IN ('1303', '1305', '1306') OR (T.CURR_CD = 'CNY' AND T.ACCT_TYP LIKE '01%'))) OR (COALESCE(T.LOAN_ACCT_BAL, 0) + COALESCE(T.INT_ADJEST_AMT, 0) > 0 AND NOT T.PENSION_INDUSTRY IS NULL AND SUBSTRING(T.ITEM_CD, 1, 6) IN ('130101', '130104'))`

# 存量个人存款信息_JS_202_CLGRCK — 取数范围

> 来源程序: `bsp_sp_js_202_clgrck.prc`

## 数据源表（FROM）

- L_PUBL_ORG_BRA_TMP AS OB
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E
- PBOCD_DATACORE.L_CODE_DICTIONARY AS F
- PBOCD_DATACORE.L_CUST_P_NEW AS C
- SMTMODS.L_CUST_C AS B
- SMTMODS.L_PUBL_RATE AS B
- SMTMODS.L_PUBL_RATE AS c

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_PUBL_RATE AS c ON A.CURR_CD = c.BASIC_CCY AND c.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND A.DATA_DATE = c.DATA_DATE AND c.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.L_CUST_C AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE AND B.DEPOSIT_CUSTTYPE IN ('13', '14')
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS F ON NVL2(B.LEGAL_CARD_NO, B.LEGAL_CARD_TYPE, B.ID_TYPE) = F.L_CODE AND F.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON A.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = A.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_DATACORE.L_CUST_P_NEW AS C ON A.CUST_ID = C.CUST_ID AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS B ON A.CURR_CD = B.BASIC_CCY AND B.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND A.DATA_DATE = B.DATA_DATE AND B.FORWARD_CCY = 'CNY'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS F ON C.ID_TYPE = F.L_CODE AND F.CODE_CLMN_NAME = 'ID_TYPE'

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_202_CLGRCK_TMP'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `A.DATA_DATE = IS_DATE`
4. `A.ACCT_BALANCE <> 0`
5. `A.GL_ITEM_CODE = '20110201' OR A.GL_ITEM_CODE = '22410101' OR A.GL_ITEM_CODE IN ('20110202', '20110203') OR A.ACCT_TYPE IN ('0401', '0402') OR A.GL_ITEM_CODE IN ('20110204', '20110211') OR A.GL_ITEM_CODE IN ('20110209', '20110210')`
6. `A.GL_ITEM_CODE IN ('20110101') OR A.GL_ITEM_CODE IN ('22410102') OR A.GL_ITEM_CODE IN ('20110103', '20110104', '20110105', '20110106', '20110107', '20110108', '20110109') OR A.GL_ITEM_CODE = '20110110' OR A.GL_ITEM_CODE IN ('20110114', '20110115') OR A.GL_ITEM_CODE = '20110102' OR A.GL_ITEM_CODE = '20110113'`
7. `DATA_DATE = IS_DATE`

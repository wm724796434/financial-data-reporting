# 存量同业存款信息_JS_202_CLTYCK — 取数范围

> 来源程序: `bsp_sp_js_202_cltyck.prc`

## 数据源表（FROM）

- (SELECT * FROM (SELECT ECIF_CUST_ID, FINA_CODE_NEW, LEGAL_FLAG, ROW_NUMBER() OVER (PARTITION BY ECIF_CUST_ID ORDER BY LEGAL_FLAG DESC NULLS FIRST, FINA_CODE_NEW NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS TY WHERE DATA_DATE = IS_DATE) AS TY WHERE TY.RN = 1) AS TY
- (SELECT DISTINCT B.DEP_ACC_CODE, B.CONT_PARTY_CODE, B.CUST_NAME, B.CONT_PARTY_TYPE, B.DEPOSIT_RESERVE_METHOD FROM PBOCD_JS_202_CLTYCK_SQ AS B WHERE B.CJRQ = VS_LAST_TEXT) AS BK
- (SELECT DISTINCT T.CUST_NAME, T.CUST_ID_NO, T.CUST_NAME_SOURCE, T.CONT_PARTY_TYPE FROM JS_102_TYKHXX_CODE AS T) AS CD1
- JS_102_TYKHXX_CODE AS CD
- L_CUST_C_TMP AS T1
- L_PUBL_ORG_BRA_TMP AS OB
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E
- PBOCD_DATACORE.L_CODE_DICTIONARY AS E2
- SMTMODS.L_AGRE_OTHER_SUBJECT_INFO AS T1
- SMTMODS.L_CUST_ALL AS B
- SMTMODS.L_PUBL_RATE AS T3

## 关联条件（JOIN）

- INNER JOIN L_CUST_C_TMP AS T1 ON T.CUST_ID = T1.CUST_ID AND T1.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS B ON T.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS T3 ON T3.BASIC_CCY = T.CURR_CD AND T3.FORWARD_CCY = 'CNY' AND T3.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN (SELECT DISTINCT B.DEP_ACC_CODE, B.CONT_PARTY_CODE, B.CUST_NAME, B.CONT_PARTY_TYPE, B.DEPOSIT_RESERVE_METHOD FROM PBOCD_JS_202_CLTYCK_SQ AS B WHERE B.CJRQ = VS_LAST_TEXT) AS BK ON T.ACCT_NUM = BK.DEP_ACC_CODE
- INNER JOIN JS_102_TYKHXX_CODE AS CD ON (CASE WHEN B.INLANDORRSHORE_FLG = 'Y' THEN CASE WHEN (T1.FINA_CODE <> 'I20000' OR T1.FINA_CODE IS NULL) THEN B.ID_NO WHEN T1.FINA_CODE = 'I20000' THEN T1.SPECIAL_CODE END WHEN B.INLANDORRSHORE_FLG = 'N' THEN COALESCE(T1.LEI_CODE, B.ID_NO, B.CUST_ID) END) = CD.CUST_ID_NO_SOURCE
- INNER JOIN (SELECT DISTINCT T.CUST_NAME, T.CUST_ID_NO, T.CUST_NAME_SOURCE, T.CONT_PARTY_TYPE FROM JS_102_TYKHXX_CODE AS T) AS CD1 ON T1.CUST_NAM = CD1.CUST_NAME_SOURCE
- INNER JOIN (SELECT * FROM (SELECT ECIF_CUST_ID, FINA_CODE_NEW, LEGAL_FLAG, ROW_NUMBER() OVER (PARTITION BY ECIF_CUST_ID ORDER BY LEGAL_FLAG DESC NULLS FIRST, FINA_CODE_NEW NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS TY WHERE DATA_DATE = IS_DATE) AS TY WHERE TY.RN = 1) AS TY ON T.CUST_ID = TY.ECIF_CUST_ID
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E ON T.ACCT_STS = E.L_CODE AND E.CODE_CLMN_NAME = 'ACCT_STS'
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS E2 ON TRIM(TY.FINA_CODE_NEW) = E2.L_CODE AND E2.CODE_CLMN_NAME = 'FINA_CODE_NEW'
- INNER JOIN SMTMODS.L_AGRE_OTHER_SUBJECT_INFO AS T1 ON T.CDS_NO = T1.SUBJECT_CD AND T.DATA_DATE = T1.DATA_DATE

## 筛选条件（WHERE）

1. `SUBSTRING(T.acct_typ, 1, 3) IN ('101', '201')`
2. `T.DATA_DATE = IS_DATE`
3. `T.BALANCE > 0`
4. `T.BALANCE * T3.CCY_RATE > 0`
5. `NOT T.ACCT_NUM IN ('9029801141140000001_1', '9019804011390200001_1')`
6. `TRIM(T.DATA_DATE) = IS_DATE`
7. `SUBSTRING(GL_ITEM_CODE, 1, 6) IN ('250202') OR GL_ITEM_CODE IN ('11010105', '15030105')`
8. `T.PRINCIPAL_BALANCE > 0`
9. `ROUND(T.PRINCIPAL_BALANCE * T3.CCY_RATE) > 0`
10. `T.DATE_SOURCESD = '存单投资'`
11. `T.FACE_VAL * T1.ISSUER_PRICE > 0`
12. `T.DATE_SOURCESD = '存单发行'`

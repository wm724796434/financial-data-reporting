# 存量再贴现信息_JS_205_CLZTX — 取数范围

> 来源程序: `bsp_sp_js_205_clztx.prc`

## 数据源表（FROM）

- (SELECT * FROM (SELECT A.*, ROW_NUMBER() OVER (PARTITION BY TYSHXYDM ORDER BY FINA_ORG_NAME NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE DATA_DATE = IS_DATE AND LEGAL_FLAG = 'Y' AND NOT TYSHXYDM IS NULL AND TYSHXYDM <> '000000000000000000' AND FINA_ORG_NAME NOT LIKE '%存托%' AND FINA_ORG_NAME NOT LIKE '%资管%' AND FINA_ORG_NAME NOT LIKE '%禁用%') AS A WHERE A.RN = 1) AS B
- (SELECT * FROM (SELECT A.CUST_ID, A.FINA_ORG_CODE, A.FINA_ORG_NAME, B.FINA_ORG_NAME AS FINA_ORG_NAME_FR, B.LEGAL_TYSHXYDM AS LEGAL_TYSHXYDM_FR, ROW_NUMBER() OVER (PARTITION BY A.FINA_ORG_NAME, A.FINA_ORG_CODE ORDER BY A.FINA_ORG_NAME NULLS LAST) AS RN FROM (SELECT * FROM (SELECT CUST_ID, FINA_ORG_CODE, FINA_ORG_NAME, TYSHXYDM, LEGAL_TYSHXYDM, ROW_NUMBER() OVER (PARTITION BY FINA_ORG_NAME, FINA_ORG_CODE ORDER BY TYSHXYDM NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE A.DATA_DATE = IS_DATE) AS A WHERE A.RN = 1) AS A LEFT
- (SELECT F.*, ROW_NUMBER() OVER (PARTITION BY f.acct_num ORDER BY f.drawdown_dt DESC NULLS FIRST) AS RN FROM SMTMODS.L_ACCT_LOAN AS f WHERE F.DATA_DATE = IS_DATE) AS F
- L_PUBL_ORG_BRA_TMP AS OB
- SMTMODS.L_AGRE_BILL_INFO AS B
- SMTMODS.L_PUBL_RATE AS R
- SMTMODS.L_PUBL_RATE AS Z

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_AGRE_BILL_INFO AS B ON A.BILL_NUM = B.BILL_NUM AND B.DATA_DATE = IS_DATE
- INNER JOIN (SELECT F.*, ROW_NUMBER() OVER (PARTITION BY f.acct_num ORDER BY f.drawdown_dt DESC NULLS FIRST) AS RN FROM SMTMODS.L_ACCT_LOAN AS f WHERE F.DATA_DATE = IS_DATE) AS F ON A.BILL_NUM = F.acct_num AND F.RN = 1 AND F.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS Z ON Z.DATA_DATE = IS_DATE AND Z.BASIC_CCY = TRIM(A.CURR_CD) AND Z.FORWARD_CCY = 'CNY' AND Z.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS R ON R.DATA_DATE = IS_DATE AND R.BASIC_CCY = TRIM(B.CURR_CD) AND R.FORWARD_CCY = 'CNY' AND R.DATA_DATE = IS_DATE
- INNER JOIN (SELECT * FROM (SELECT A.CUST_ID, A.FINA_ORG_CODE, A.FINA_ORG_NAME, B.FINA_ORG_NAME AS FINA_ORG_NAME_FR, B.LEGAL_TYSHXYDM AS LEGAL_TYSHXYDM_FR, ROW_NUMBER() OVER (PARTITION BY A.FINA_ORG_NAME, A.FINA_ORG_CODE ORDER BY A.FINA_ORG_NAME NULLS LAST) AS RN FROM (SELECT * FROM (SELECT CUST_ID, FINA_ORG_CODE, FINA_ORG_NAME, TYSHXYDM, LEGAL_TYSHXYDM, ROW_NUMBER() OVER (PARTITION BY FINA_ORG_NAME, FINA_ORG_CODE ORDER BY TYSHXYDM NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE A.DATA_DATE = IS_DATE) AS A WHERE A.RN = 1) AS A LEFT JOIN (SELECT * FROM (SELECT A.*, ROW_NUMBER() OVER (PARTITION BY TYSHXYDM ORDER BY FINA_ORG_NAME NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE DATA_DATE = IS_DATE AND LEGAL_FLAG = 'Y' AND NOT TYSHXYDM IS NULL AND TYSHXYDM <> '000000000000000000' AND FINA_ORG_NAME NOT LIKE '%存托%' AND FINA_ORG_NAME NOT LIKE '%资管%' AND FINA_ORG_NAME NOT LIKE '%禁用%') AS A WHERE A.RN = 1) AS B ON A.LEGAL_TYSHXYDM = B.TYSHXYDM) WHERE RN = 1) AS FR ON B.PAY_BANK_ID = FR.FINA_ORG_CODE
- INNER JOIN (SELECT * FROM (SELECT A.*, ROW_NUMBER() OVER (PARTITION BY TYSHXYDM ORDER BY FINA_ORG_NAME NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE DATA_DATE = IS_DATE AND LEGAL_FLAG = 'Y' AND NOT TYSHXYDM IS NULL AND TYSHXYDM <> '000000000000000000' AND FINA_ORG_NAME NOT LIKE '%存托%' AND FINA_ORG_NAME NOT LIKE '%资管%' AND FINA_ORG_NAME NOT LIKE '%禁用%') AS A WHERE A.RN = 1) AS B ON A.LEGAL_TYSHXYDM = B.TYSHXYDM
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = TRIM(T.ORG_NUM) AND OB.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `TABLE_NAME = 'JS_205_CLZTX'`
2. `PARTITION_NAME = 'JS_205_CLZTX_' || IS_DATE`
3. `A.DATA_DATE = IS_DATE`
4. `A.ACCT_TYP IN ('20303', '20304')`
5. `A.BALANCE > 0`
6. `TRIM(T.DATA_DATE) = IS_DATE`

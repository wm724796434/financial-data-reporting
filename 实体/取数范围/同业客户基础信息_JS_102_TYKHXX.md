# 同业客户基础信息_JS_102_TYKHXX — 取数范围

> 来源程序: `bsp_sp_js_102_tykhxx.prc`

## 数据源表（FROM）

- JS_102_TYKHXX_ALL AS BK
- JS_102_TYKHXX_CODE AS CD
- L_CUST_C_TMP AS B
- L_CUST_C_TMP AS B1
- M_EAST_META_FIELD_SCOPE AS PJ
- TMP_JS_102_FTYKHX_GLF AS GLF

## 关联条件（JOIN）

- INNER JOIN L_CUST_C_TMP AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE
- INNER JOIN JS_102_TYKHXX_CODE AS CD ON B.ID_NO = CD.CUST_ID_NO_SOURCE AND B.CUST_NAM = CD.CUST_NAME_SOURCE
- INNER JOIN L_CUST_C_TMP AS B1 ON CD.CUST_ID = B1.CUST_ID AND B1.DATA_DATE = IS_DATE
- INNER JOIN TMP_JS_102_FTYKHX_GLF AS GLF ON NVL2(B1.CUST_ID, B1.CUST_NAM, B.CUST_NAM) = GLF.CUST_NAME
- INNER JOIN M_EAST_META_FIELD_SCOPE AS PJ ON NVL2(B1.CUST_ID, B1.CUS_RISK_LV_DE, B.CUS_RISK_LV_DE) = PJ.PARAM_CODE AND PJ.PARAM_TYPE = 'NBPJ'
- INNER JOIN JS_102_TYKHXX_ALL AS BK ON A.CUST_ID_NO = BK.CUST_ID_NO AND A.FRNBJGH = BK.FRNBJGH

## 筛选条件（WHERE）

1. `TRIM(A.DATA_DATE) = IS_DATE`
2. `A.BALANCE <> 0`
3. `TRIM(A.DATA_DATE) = VS_TEXT`
4. `NOT A.CUST_ID IN (SELECT CUST_ID FROM JS_102_TYKHXX_TEMP1)`
5. `TABLE_NAME = 'JS_102_TYKHXX'`
6. `PARTITION_NAME = 'P' || IS_DATE`
7. `T.RN = 1`
8. `NOT T.CUST_ID_NO IS NULL`
9. `NOT EXISTS(SELECT 1 FROM PBOCD_JS_102_TYKHXX AS B WHERE B.CJRQ = IS_DATE AND A.CUST_ID_NO = B.CUST_ID_NO AND A.FRNBJGH = B.FRNBJGH)`

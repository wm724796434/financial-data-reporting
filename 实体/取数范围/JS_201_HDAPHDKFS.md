# JS_201_HDAPHDKFS — 取数范围

- A_REPT_ITEM_VAL_A3302_TMP3 AS B
- L_PUBL_ORG_BRA_TMP AS OB
- PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_LOAN AS B
- PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_LOAN AS CYDB
- PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_TMP1 AS GTJY
- PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_TMP1 AS NHJY
- PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_TMP1 AS XW

- SMTMODS.L_CUST_C AS C1

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_LOAN AS T1 ON T1.DATA_DATE = IS_DATE AND T.ACCT_NUM || T.DRAFT_RNG = T1.ACCT_NUM || T1.DRAFT_RNG AND SUBSTRING(T1.ITEM_CD, 1, 6) IN ('130101', '130104') AND SUBSTRING(T.ITEM_CD, 1, 6) IN ('130102', '130105')
- INNER JOIN SMTMODS.L_CUST_C AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE AND SUBSTRING(B.CUST_TYP, 1, 1) IN ('0', '1') OR B.CUST_TYP = '9101' AND B.CUST_TYP <> '14' AND B.CORP_SCALE IN ('S', 'T')
- INNER JOIN SMTMODS.AGRE_CREDITLINE_INFO AS C ON C.DATA_DATE = IS_DATE AND A.CUST_ID = C.CUST_ID AND C.FACILITY_AMT <= 10000000
- INNER JOIN A_REPT_ITEM_VAL_A3302_TMP3 AS B ON A.CUST_ID = B.CUST_ID
- INNER JOIN PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_LOAN AS B ON A.LOAN_NUM = B.LOAN_NUM AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.AGRE_CREDITLINE_INFO AS C ON C.DATA_DATE = IS_DATE AND A.CUST_ID = C.CUST_ID AND C.FACILITY_AMT <= 5000000
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.DATA_DATE = IS_DATE AND U.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND U.BASIC_CCY = a.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.DATA_DATE = IS_DATE AND U.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND U.BASIC_CCY = B.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.DATA_DATE = IS_DATE AND U.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND U.BASIC_CCY = A.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_TMP1 AS XW ON XW.LOAN_NUM = T.LOAN_NUM AND XW.FLAG = '1XW'
- INNER JOIN PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_TMP1 AS GTJY ON GTJY.LOAN_NUM = T.LOAN_NUM AND GTJY.FLAG = '2GTJY'
- INNER JOIN PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_TMP1 AS XWJY ON XWJY.LOAN_NUM = T.LOAN_NUM AND XWJY.FLAG = '3XWJY'
- INNER JOIN PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_TMP1 AS NHJY ON NHJY.LOAN_NUM = T.LOAN_NUM AND NHJY.FLAG = '4NHJY'
- INNER JOIN PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_LOAN AS CYDB ON CYDB.LOAN_NUM = T.LOAN_NUM AND CYDB.UNDERTAK_GUAR_TYPE <> '#' AND NOT CYDB.UNDERTAK_GUAR_TYPE IS NULL AND CYDB.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS C1 ON T.CUST_ID = C1.CUST_ID AND C1.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS A2 ON A2.DATA_DATE = IS_DATE AND A2.CUST_ID = T.CUST_ID
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.DATA_DATE = IS_DATE AND U.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND U.BASIC_CCY = T.CURR_CD AND U.FORWARD_CCY = 'CNY'

## 筛选条件（WHERE）

1. `T.DATA_DATE = IS_DATE`
2. `CAST(T.DRAWDOWN_DT AS TEXT) = SUBSTRING(IS_DATE, 1, 6)`
3. `T.OPERATE_CUST_TYPE IN ('A', 'B')`
4. `A.DATA_DATE = IS_DATE`
5. `CAST(A.DRAWDOWN_DT AS TEXT) = SUBSTRING(IS_DATE, 1, 6)`
6. `NOT A.ACCT_TYP IN ('B01', 'C01', 'D01', '90')`
7. `NOT A.LOAN_NUM IN (SELECT TT.Loan_Num FROM SMTMODS.L_ACCT_TRAD_FIN AS TT WHERE TT.DATA_DATE = IS_DATE AND TT.TRAD_FIN_TYPE IN ('12', '13') AND TT.FORF_TYPE IN ('2'))`
8. `A.ACCT_TYP IN ('0102', '010201', '010299')`
9. `B.ACCT_TYP NOT LIKE '90%'`
10. `CAST(B.DRAWDOWN_DT AS TEXT) = SUBSTRING(IS_DATE, 1, 6)`
11. `SUBSTRING(A.SNDKFL, 1, 5) IN ('P_101', 'P_103')`
12. `b.acct_typ LIKE '0102%'`
13. `B.ACCT_TYP LIKE '0102%'`
14. `SUBSTRING(A.ITEM_CD, 1, 4) IN ('1303', '1305', '1306') OR (A.CURR_CD = 'CNY' AND A.ACCT_TYP LIKE '01%') OR (COALESCE(A.LOAN_ACCT_BAL, 0) + COALESCE(A.INT_ADJEST_AMT, 0) > 0 AND SUBSTRING(A.ITEM_CD, 1, 6) IN ('130101', '130104'))`
15. `(A.UNDERTAK_GUAR_TYPE <> '#' AND NOT A.UNDERTAK_GUAR_TYPE IS NULL) OR A.ACCT_TYP = '0104'`
16. `NOT XW.LOAN_NUM IS NULL OR NOT GTJY.LOAN_NUM IS NULL OR NOT XWJY.LOAN_NUM IS NULL OR NOT NHJY.LOAN_NUM IS NULL OR NOT CYDB.LOAN_NUM IS NULL`
17. `SUBSTRING(T.ITEM_CD, 1, 4) IN ('1303', '1305', '1306') OR (T.CURR_CD = 'CNY' AND T.ACCT_TYP LIKE '01%') OR (COALESCE(T.LOAN_ACCT_BAL, 0) + COALESCE(T.INT_ADJEST_AMT, 0) > 0 AND SUBSTRING(t.ITEM_CD, 1, 6) IN ('130101', '130104'))`
18. `NOT EXISTS(SELECT 1 FROM PBOCD_DATACORE.PBOCD_JS_201_HDAPHDKFS_LOAN AS A2 WHERE A2.DATA_DATE = IS_DATE AND SUBSTRING(A2.ITEM_CD, 1, 4) IN ('1306') AND SUBSTRING(CAST(A2.DRAWDOWN_DT AS TEXT), 1, 6) = SUBSTRING(IS_DATE, 1, 6) AND A2.DRAWDOWN_DT = A2.FINISH_DT AND A2.LOAN_ACCT_BAL = 0 AND T.LOAN_NUM = A2.LOAN_NUM)`

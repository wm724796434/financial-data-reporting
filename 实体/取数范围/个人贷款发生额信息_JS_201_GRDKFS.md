# 个人贷款发生额信息_JS_201_GRDKFS — 取数范围

> 来源程序: `bsp_sp_js_201_grdkfs.prc`

## 数据源表（FROM）

- (SELECT ROW_NUMBER() OVER (PARTITION BY T.CUST_ID ORDER BY T.CUST_ID DESC NULLS FIRST) AS RN, T.* FROM JS_102_GRKHXX AS T WHERE T.DATA_DATE = IS_DATE) AS KH
- (SELECT T.LOAN_NUM, ROW_NUMBER() OVER (PARTITION BY T.CUST_ID ORDER BY T.DRAWDOWN_DT ASC NULLS LAST, LOAN_NUM NULLS LAST) AS RN FROM SMTMODS.L_ACCT_LOAN AS T WHERE t.data_date = IS_DATE) AS LA
- DBFS_TMP AS TP7
- JS_201_GRDKFS_TEMP02 AS T2
- L_ACCT_LOAN_SUOQI AS T1
- L_PUBL_ORG_BRA_TMP AS C
- L_PUBL_ORG_BRA_TMP AS OB
- M_DICT_REMAPPING AS X
- PBOCD_JS_102_GRKHXX AS AA
- PBOCD_JS_201_CLGRDK_SQ AS BK
- PBOCD_JS_201_CLGRDK_SQ AS SQ
- SMTMODS.L_ACCT_LOAN AS F
- SMTMODS.L_CUST_ALL AS B
- SMTMODS.L_CUST_C AS E
- SMTMODS.L_CUST_IDENTIFY AS D
- SMTMODS.L_PUBL_RATE AS U

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_CUST_ALL AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS C ON A.ORG_NUM = C.ORG_NUM AND C.DATA_DATE = IS_DATE
- INNER JOIN (SELECT ROW_NUMBER() OVER (PARTITION BY T.CUST_ID ORDER BY T.CUST_ID DESC NULLS FIRST) AS RN, T.* FROM JS_102_GRKHXX AS T WHERE T.DATA_DATE = IS_DATE) AS KH ON A.CUST_ID = KH.CUST_ID AND KH.DATA_DATE = IS_DATE AND KH.RN = 1
- INNER JOIN SMTMODS.L_CUST_IDENTIFY AS D ON B.CUST_ID = D.CUST_ID AND B.ID_TYPE = D.ID_TYPE AND B.ID_NO = D.ID_NO AND D.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS E ON A.CUST_ID = E.CUST_ID AND E.DATA_DATE = IS_DATE
- INNER JOIN M_DICT_REMAPPING AS X ON B.NATION_CD1 = X.ORI_VALUES AND X.DICT_CODE = 'COUNTRY_CODE_3_TO_NUM'
- INNER JOIN DBFS_TMP AS TP7 ON a.LOAN_NUM = TP7.LOAN_NUM
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.CCY_DATE = STR_TO_DATE(CAST(A.DRAWDOWN_DT AS TEXT), '%Y%m%d') AND U.BASIC_CCY = A.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN (SELECT T.LOAN_NUM, ROW_NUMBER() OVER (PARTITION BY T.CUST_ID ORDER BY T.DRAWDOWN_DT ASC NULLS LAST, LOAN_NUM NULLS LAST) AS RN FROM SMTMODS.L_ACCT_LOAN AS T WHERE t.data_date = IS_DATE) AS LA ON A.LOAN_NUM = LA.LOAN_NUM
- INNER JOIN L_ACCT_LOAN_SUOQI AS T1 ON A.LOAN_NUM = T1.LOAN_NUM AND A.MATURITY_DT_BEFORE = T1.MATURITY_DT_BEFORE AND A.MATURITY_DT = T1.MATURITY_DT
- INNER JOIN SMTMODS.L_ACCT_LOAN AS F ON A.LOAN_NUM = F.LOAN_NUM AND F.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS B ON F.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE
- INNER JOIN (SELECT ROW_NUMBER() OVER (PARTITION BY T.CUST_ID ORDER BY T.CUST_ID DESC NULLS FIRST) AS RN, T.* FROM JS_102_GRKHXX AS T WHERE T.DATA_DATE = IS_DATE) AS KH ON F.CUST_ID = KH.CUST_ID AND KH.DATA_DATE = IS_DATE AND KH.RN = 1
- INNER JOIN SMTMODS.L_CUST_C AS E ON F.CUST_ID = E.CUST_ID AND E.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.CCY_DATE = STR_TO_DATE(CAST(A.REPAY_DT AS TEXT), '%Y%m%d') AND U.BASIC_CCY = F.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN PBOCD_JS_201_CLGRDK_SQ AS SQ ON a.LOAN_NUM = SQ.LOAN_NUM AND SQ.CJRQ = VS_LAST_TEXT
- INNER JOIN L_ACCT_LOAN_SUOQI AS T1 ON F.LOAN_NUM = T1.LOAN_NUM AND F.MATURITY_DT_BEFORE = T1.MATURITY_DT_BEFORE AND F.MATURITY_DT = T1.MATURITY_DT
- INNER JOIN JS_201_GRDKFS_TEMP02 AS T2 ON T.SERIAL_NO = T2.SERIAL_NO
- INNER JOIN PBOCD_JS_201_CLGRDK_SQ AS BK ON T.LOAN_NUM = BK.LOAN_NUM AND BK.CJRQ = VS_LAST_TEXT
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.NBJGH AND OB.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_JS_102_GRKHXX AS AA ON T.CUST_ID_NO = AA.CUST_ID_NO AND AA.CJRQ = IS_DATE AND AA.FRNBJGH = '990000'

## 筛选条件（WHERE）

1. `TABLE_NAME = 'JS_201_GRDKFS'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `DATE_TRUNC('MM', A.DRAWDOWN_DT) = DATE_TRUNC('MM', D_DATADATE) OR ((A.INTERNET_LOAN_FLG = 'Y' OR A.CP_ID = 'DK001000100041') AND A.DRAWDOWN_DT = (DATE_TRUNC('MM', STR_TO_DATE(IS_DATE, '%Y%m%d')) - 1))`
4. `C.NATION_CD = 'CHN'`
5. `SUBSTRING(A.ACCT_TYP, 1, 2) IN ('01', '02', '04', '05', '08') OR A.ACCT_TYP = '070101'`
6. `B.CUST_TYPE = '00' OR E.CUST_TYP = '3'`
7. `A.DATA_DATE = IS_DATE`
8. `A.CANCEL_FLG = 'N'`
9. `A.LOAN_STOCKEN_DATE IS NULL`
10. `DATE_TRUNC('MM', A.REPAY_DT) = DATE_TRUNC('MM', D_DATADATE) OR ((F.INTERNET_LOAN_FLG = 'Y' OR F.CP_ID = 'DK001000100041') AND A.REPAY_DT = (DATE_TRUNC('MM', STR_TO_DATE(IS_DATE, '%Y%m%d')) - 1))`
11. `DATE_TRUNC('MM', STR_TO_DATE(A.DATA_DATE, '%Y%m%d')) = DATE_TRUNC('MM', STR_TO_DATE(IS_DATE, '%Y%m%d'))`
12. `A.PAY_AMT <> 0`
13. `SUBSTRING(F.ACCT_TYP, 1, 2) IN ('01', '02', '04', '05', '08') OR F.ACCT_TYP = '070101'`
14. `F.CANCEL_FLG = 'N' AND F.LOAN_STOCKEN_DATE IS NULL OR (F.CANCEL_FLG = 'Y' AND EXISTS(SELECT 1 FROM SMTMODS.L_ACCT_WRITE_OFF AS XX WHERE XX.DATA_DATE = IS_DATE AND SUBSTRING(CAST(XX.WRITE_OFF_DATE AS TEXT), 1, 6) = SUBSTRING(IS_DATE, 1, 6) AND A.LOAN_NUM = XX.LOAN_NUM))`
15. `TRIM(T.DATA_DATE) = IS_DATE`
16. `TRIM(T.DATA_DATE) = VS_TEXT`
17. `T.TRANS_AMT <> 0`
18. `T.DATA_DATE = IS_DATE`
19. `CJRQ = IS_DATE`

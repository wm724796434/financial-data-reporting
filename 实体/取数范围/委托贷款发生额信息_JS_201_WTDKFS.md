## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_LOAN AS T ON M.LOAN_NUM = T.LOAN_NUM AND T.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS LC ON T.CUST_ID = LC.CUST_ID AND LC.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS LP ON T.CUST_ID = LP.CUST_ID AND LP.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_Cust_ALL AS H ON T.CUST_ID = H.CUST_ID AND H.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS CD1 ON LC.id_type = CD1.L_CODE AND CD1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD2 ON LP.ID_TYPE = CD2.L_CODE AND CD2.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD3 ON LC.CORP_HOLD_TYPE = CD3.L_CODE AND CD3.CODE_CLMN_NAME = 'CORP_HOLD_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS D4 ON TRIM(H.DEPT_TYPE) = D4.L_CODE AND D4.CODE_CLMN_NAME = 'DEPT_TYPE'
- INNER JOIN SMTMODS.L_ACCT_LOAN_ENTRUST AS T1 ON T.LOAN_NUM = T1.LOAN_NUM AND T1.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_Cust_ALL AS H ON H.DATA_DATE = IS_DATE AND T1.TRUSTOR_ID = H.CUST_ID
- INNER JOIN L_CUST_C_TMP AS LC ON T1.TRUSTOR_ID = LC.CUST_ID AND LC.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS LP ON T1.TRUSTOR_ID = LP.CUST_ID AND LP.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS CD1 ON LC.ID_TYPE = CD1.L_CODE AND CD1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN JS_201_WTDKFS_TEMP04 AS T1 ON T.JJBH = T1.LOAN_NUM
- INNER JOIN JS_201_WTDKFS_TEMP05 AS T2 ON T.JJBH = T2.LOAN_NUM
- INNER JOIN SMTMODS.L_ACCT_LOAN AS T3 ON T.JJBH = T3.LOAN_NUM AND T3.DATA_DATE = IS_DATE
- INNER JOIN DBFS_TMP AS TP7 ON T.JJBH = TP7.LOAN_NUM
- INNER JOIN SMTMODS.L_ACCT_LOAN_ENTRUST AS T4 ON T3.LOAN_NUM = T4.LOAN_NUM AND T4.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.NBJGH AND OB.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_JS_201_CLWTDK_SQ AS BK ON T.LOAN_NUM = BK.LOAN_NUM AND BK.CJRQ = VS_LAST_TEXT

## 筛选条件（WHERE）

1. `TABLE_NAME = 'JS_201_WTDKFS'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `SUBSTRING(CAST(T.DRAWDOWN_DT AS TEXT), 1, 6) = SUBSTRING(IS_DATE, 1, 6)`
4. `T.ITEM_CD LIKE '3020%'`
5. `T.CANCEL_FLG = 'N'`
6. `NOT T.ITEM_CD IN ('30200201', '30200202')`
7. `T.LOAN_STOCKEN_DATE IS NULL`
8. `M.PAY_AMT <> 0`
9. `SUBSTRING(CAST(M.REPAY_DT AS TEXT), 0, 6) = SUBSTRING(IS_DATE, 0, 6)`
10. `T.DATA_DATE = IS_DATE`
11. `t.data_date = is_date`
12. `TRIM(T.DATA_DATE) = IS_DATE`

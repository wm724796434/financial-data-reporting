# 担保合同信息_JS_201_DBHTXX — 取数范围

> 来源程序: `bsp_sp_js_201_dbhtxx.prc`

## 数据源表（FROM）

- (SELECT * FROM SMTMODS.L_AGRE_GUARANTY_INFO WHERE DATA_DATE = IS_DATE AND COLL_STATUS = 'Y') AS S2
- (SELECT * FROM SMTMODS.L_AGRE_GUA_RELATION AS T1 WHERE T1.DATA_DATE = IS_DATE AND REL_STATUS = 'Y') AS T1
- (SELECT ACCT_NUM, LOAN_NUM, DEPARTMENTD, ROW_NUMBER() OVER (PARTITION BY acct_num ORDER BY loan_num DESC NULLS FIRST) AS rn FROM SMTMODS.L_ACCT_LOAN WHERE data_date = IS_DATE) AS T9
- (SELECT DISTINCT T.GUAR_CONTRACT_NUM, t1.CONTRACT_NUM, CASE WHEN T.COLL_VALUE_SUM = '0' THEN NULL ELSE (T2.LOAN_ACCT_BAL_SUM / T.COLL_VALUE_SUM) END AS DZYL, T.COLL_VALUE_SUM AS COLL_VALUE FROM DBHTXX_PLED_TEMP1 AS T LEFT
- ACCT_LOAN_TMP01 AS T7
- DBHTXX_BZR_TMP01 AS t5
- DBHTXX_PLED_TEMP2 AS T2
- JS_102_FTYKHX AS F
- JS_102_FTYKHX_TEMP03 AS T7
- JS_201_DBHTXX_BL AS t2
- JS_201_DBHTXX_BL AS t3
- JS_201_DBHTXX_MAPPING AS t1
- L_CODE_DICTIONARY AS D1
- L_CODE_DICTIONARY AS D2
- L_CODE_DICTIONARY AS D4
- L_CUST_C_tmp AS G
- L_PUBL_ORG_BRA_TMP AS OB
- L_PUBL_ORG_BRA_TMP AS OB2
- PBOCD_JS_201_DBHTXX_SQ AS t4
- SMTMODS.L_AGRE_GUARANTEE_CONTRACT AS T
- SMTMODS.L_AGRE_GUARANTEE_RELATION AS GR
- SMTMODS.L_AGRE_LOAN_CONTRACT AS T4
- SMTMODS.L_CODE_DICTIONARY AS D3
- SMTMODS.L_CUST_ALL AS M
- SMTMODS.L_CUST_C AS C
- SMTMODS.L_CUST_C AS T8
- SMTMODS.L_CUST_P AS P
- SMTMODS.L_PUBL_RATE AS T2

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_AGRE_GUARANTEE_RELATION AS GR ON GI.GUARANTEE_SERIAL_NUM = GR.GUARANTEE_SERIAL_NUM AND GR.DATA_DATE = IS_DATE AND GR.REL_STATUS = 'Y'
- INNER JOIN SMTMODS.L_AGRE_GUARANTEE_CONTRACT AS T ON T2.GUAR_CONTRACT_NUM = T.GUAR_CONTRACT_NUM AND T.DATA_DATE = IS_DATE AND SUBSTRING(T.GUAR_TYP, 1, 1) = 'C'
- INNER JOIN JS_102_FTYKHX AS F ON t2.GUAR_CUST_ID = F.CUST_ID AND F.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS C ON t2.GUAR_CUST_ID = C.CUST_ID AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS M ON t2.GUAR_CUST_ID = M.CUST_ID AND M.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS P ON t2.GUAR_CUST_ID = P.CUST_ID AND P.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS D1 ON TRIM(CASE WHEN C.CUST_TYP = 3 THEN C.LEGAL_CARD_TYPE ELSE C.ID_TYPE END) = D1.L_CODE AND D1.CODE_CLMN_NAME = 'BZR_ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS D2 ON TRIM(P.ID_TYPE) = D2.L_CODE AND D2.CODE_CLMN_NAME = 'BZR_ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS D4 ON TRIM(T2.GUARANTEE_ID_TPYE) = D4.L_CODE AND D4.CODE_CLMN_NAME = 'BZR_ID_TYPE'
- INNER JOIN L_CUST_C_tmp AS G ON t2.GUAR_CUST_ID = G.CUST_ID AND G.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CODE_DICTIONARY AS D3 ON TRIM(G.DEPT_TYPE) = D3.CODE AND D3.CODE_CLMN_NAME = 'DEPT_TYPE'
- INNER JOIN (SELECT * FROM SMTMODS.L_AGRE_GUA_RELATION AS T1 WHERE T1.DATA_DATE = IS_DATE AND REL_STATUS = 'Y') AS T1 ON T.GUAR_CONTRACT_NUM = T1.GUAR_CONTRACT_NUM
- INNER JOIN (SELECT DISTINCT T.GUAR_CONTRACT_NUM, t1.CONTRACT_NUM, CASE WHEN T.COLL_VALUE_SUM = '0' THEN NULL ELSE (T2.LOAN_ACCT_BAL_SUM / T.COLL_VALUE_SUM) END AS DZYL, T.COLL_VALUE_SUM AS COLL_VALUE FROM DBHTXX_PLED_TEMP1 AS T LEFT JOIN (SELECT * FROM SMTMODS.L_AGRE_GUA_RELATION AS T1 WHERE T1.DATA_DATE = IS_DATE AND REL_STATUS = 'Y') AS T1 ON T.GUAR_CONTRACT_NUM = T1.GUAR_CONTRACT_NUM LEFT JOIN DBHTXX_PLED_TEMP2 AS T2 ON T1.CONTRACT_NUM = T2.CONTRACT_NUM) AS T3 ON T.GUAR_CONTRACT_NUM = T3.GUAR_CONTRACT_NUM AND t1.CONTRACT_NUM = t3.CONTRACT_NUM
- INNER JOIN SMTMODS.L_AGRE_LOAN_CONTRACT AS T4 ON T1.CONTRACT_NUM = T4.CONTRACT_NUM AND T4.DATA_DATE = IS_DATE
- INNER JOIN DBHTXX_BZR_TMP01 AS t5 ON T.GUAR_CONTRACT_NUM = t5.GUAR_CONTRACT_NUM
- INNER JOIN ACCT_LOAN_TMP01 AS T7 ON T1.CONTRACT_NUM = T7.ACCT_NUM
- INNER JOIN SMTMODS.L_CUST_C AS T8 ON T4.CUST_ID = T8.CUST_ID AND T8.DATA_DATE = IS_DATE
- INNER JOIN DBHTXX_PLED_TEMP2 AS T2 ON T1.CONTRACT_NUM = T2.CONTRACT_NUM
- INNER JOIN (SELECT * FROM SMTMODS.L_AGRE_GUARANTY_INFO WHERE DATA_DATE = IS_DATE AND COLL_STATUS = 'Y') AS S2 ON S1.GUARANTEE_SERIAL_NUM = S2.GUARANTEE_SERIAL_NUM
- INNER JOIN SMTMODS.L_PUBL_RATE AS T2 ON T.CURR_CD = T2.BASIC_CCY AND T2.FORWARD_CCY = 'CNY' AND T2.DATA_DATE = IS_DATE
- INNER JOIN JS_102_FTYKHX_TEMP03 AS T7 ON T.CUST_ID = T7.CUST_ID
- INNER JOIN (SELECT ACCT_NUM, LOAN_NUM, DEPARTMENTD, ROW_NUMBER() OVER (PARTITION BY acct_num ORDER BY loan_num DESC NULLS FIRST) AS rn FROM SMTMODS.L_ACCT_LOAN WHERE data_date = IS_DATE) AS T9 ON T.CONTRACT_NUM = T9.ACCT_NUM AND T9.RN = '1'
- INNER JOIN JS_201_DBHTXX_BL AS t2 ON t.guar_con_num = t2.guar_con_num AND t.contract_code = t2.contract_code AND COALESCE(t.guar_id_no, 'kkk') = COALESCE(t2.guar_id_no, 'kkk') AND t2.opt_type = 'D'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.NBJGH AND OB.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB2 ON OB.UP_ORG_NUM = OB2.ORG_NUM AND OB.DATA_DATE = IS_DATE AND OB2.DATA_DATE = IS_DATE
- INNER JOIN JS_201_DBHTXX_MAPPING AS t1 ON t1.guar_con_num = t.guar_con_num AND t1.contract_code = t.contract_code AND COALESCE(t1.cust_id_no, 'UUU') = COALESCE(t.guar_id_no, 'UUU') OR COALESCE(t1.cust_id_no_new, 'UUU') = COALESCE(t.guar_id_no, 'UUU')
- INNER JOIN JS_201_DBHTXX_BL AS t3 ON t1.guar_con_num = t3.guar_con_num AND t1.contract_code = t3.contract_code AND COALESCE(t1.cust_id_no_new, 'UUU') = COALESCE(t3.guar_id_no, 'UUU') AND t3.opt_type = 'U'
- INNER JOIN PBOCD_JS_201_DBHTXX_SQ AS t4 ON t.guar_con_num = t4.guar_con_num AND t.contract_code = t4.contract_code AND COALESCE(t.guar_id_no, 'UUU') = COALESCE(t4.guar_id_no, 'UUU') AND t4.data_date = CAST(ADD_MONTHS(STR_TO_DATE(IS_DATE, 'yyyymmdd'), -1) AS TEXT)

## 筛选条件（WHERE）

1. `T.DATA_DATE = IS_DATE`
2. `T.CANCEL_FLG = 'Y'`
3. `GI.DATA_DATE = IS_DATE`
4. `GI.COLL_STATUS = 'Y'`
5. `D.DATA_DATE = IS_DATE`
6. `D.LOAN_ACCT_BAL > 0`
7. `D.CANCEL_FLG = 'N'`
8. `D.LOAN_STOCKEN_DATE IS NULL`
9. `t4.RN = 1`
10. `T8.CUST_TYP <> 3`
11. `T.GUAR_CONTRACT_STATUS = 'Y'`
12. `T7.LOAN_ACCT_BAL > 0`
13. `NOT T.GUAR_CONTRACT_END_DT IS NULL`
14. `EXISTS(SELECT 1 FROM SMTMODS.L_AGRE_GUARANTEE_RELATION AS S1 INNER JOIN (SELECT * FROM SMTMODS.L_AGRE_GUARANTY_INFO WHERE DATA_DATE = IS_DATE AND COLL_STATUS = 'Y') AS S2 ON S1.GUARANTEE_SERIAL_NUM = S2.GUARANTEE_SERIAL_NUM WHERE S1.DATA_DATE = IS_DATE AND S1.REL_STATUS = 'Y' AND T.GUAR_CONTRACT_NUM = S1.GUAR_CONTRACT_NUM AND T.GUAR_TYP IN ('A0101', 'B0101')) OR EXISTS(SELECT 1 FROM SMTMODS.L_AGRE_GUARANTEE_RELATION AS S1 WHERE S1.DATA_DATE = IS_DATE AND S1.REL_STATUS = 'Y' AND T.GUAR_CONTRACT_NUM = S1.GUAR_CONTRACT_NUM AND T.GUAR_TYP LIKE 'C%')`
15. `TABLE_NAME = 'JS_201_DBHTXX'`
16. `PARTITION_NAME = 'P' || IS_DATE`
17. `LENGTHB(TRUNC(COALESCE(T.COLLATERAL_RATIO, 0))) < 9`
18. `T7.CUST_ID IS NULL OR SUBSTRING(T.CONTRACT_NUM, 1, 100) IN ('100101140010104624', '012201160010266600', '010301170010357407', '010301170010360672', '010301170010363236', '010301170010364393', '010301170010364938', '010301170010364371', '010301170010364402', '82601250806150221')`
19. `t2.opt_type IS NULL`
20. `T.DATA_DATE = VS_TEXT`

# 银行承兑汇票发生额信息_JS_205_YHCDFS — 取数范围

> 来源程序: `bsp_sp_js_205_yhcdfs.prc`

## 数据源表（FROM）

- (SELECT LOAN.ACCT_NUM, MIN(LOAN.DATA_DATE) AS DATA_DATE FROM SMTMODS.L_ACCT_OBS_lOAN AS LOAN WHERE LOAN.DATA_DATE BETWEEN CAST(ADD_MONTHS(STR_TO_DATE(IS_DATE, '%Y%m%d'), -1) + 1 AS TEXT) AND IS_DATE AND LOAN.BALANCE = 0 GROUP BY LOAN.ACCT_NUM) AS LOAN
- (SELECT T.CONTRACT_NUM, COUNT(DISTINCT SUBSTRING(T1.GUAR_TYP, 1, 1)) AS CN FROM SMTMODS.L_AGRE_GUA_RELATION AS T INNER
- JS_205_CLYHCD_TMP2 AS DBW
- L_CODE_DICTIONARY AS CD1
- L_CODE_DICTIONARY AS CD2
- L_CODE_DICTIONARY AS CD4
- L_PUBL_ORG_BRA_TMP AS OB
- SMTMODS.L_ACCT_LOAN AS C
- SMTMODS.L_AGRE_BILL_INFO AS B
- SMTMODS.L_AGRE_GUARANTEE_CONTRACT AS T1
- SMTMODS.L_AGRE_LOAN_CONTRACT AS H
- SMTMODS.L_CUST_C AS F
- SMTMODS.L_CUST_P AS D
- SMTMODS.L_PUBL_RATE AS R

## 关联条件（JOIN）

- INNER JOIN (SELECT LOAN.ACCT_NUM, MIN(LOAN.DATA_DATE) AS DATA_DATE FROM SMTMODS.L_ACCT_OBS_lOAN AS LOAN WHERE LOAN.DATA_DATE BETWEEN CAST(ADD_MONTHS(STR_TO_DATE(IS_DATE, '%Y%m%d'), -1) + 1 AS TEXT) AND IS_DATE AND LOAN.BALANCE = 0 GROUP BY LOAN.ACCT_NUM) AS LOAN ON A.BILL_NUM = LOAN.ACCT_NUM
- INNER JOIN SMTMODS.L_ACCT_LOAN AS C ON C.DATA_DATE = IS_DATE AND 'DK' || A.BILL_NUM = C.LOAN_NUM
- INNER JOIN SMTMODS.L_PUBL_RATE AS R ON R.DATA_DATE = IS_DATE AND R.BASIC_CCY = TRIM(C.CURR_CD) AND R.FORWARD_CCY = 'CNY' AND R.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_AGRE_BILL_INFO AS B ON A.ACCT_NUM = B.BILL_NUM AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS D ON A.CUST_ID = D.CUST_ID AND D.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS F ON A.CUST_ID = F.CUST_ID AND F.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_AGRE_LOAN_CONTRACT AS H ON A.ACCT_NO = H.CONTRACT_NUM AND H.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS CD1 ON D.ID_TYPE = CD1.L_CODE AND CD1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD2 ON F.ID_TYPE = CD2.L_CODE AND CD2.CODE_CLMN_NAME = 'BZR_ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD4 ON F.CORP_HOLD_TYPE = CD4.L_CODE AND CD4.CODE_CLMN_NAME = 'CORP_HOLD_TYPE'
- INNER JOIN SMTMODS.L_ACCT_LOAN AS C ON C.DATA_DATE = IS_DATE AND 'DK' || B.BILL_NUM = C.LOAN_NUM
- INNER JOIN SMTMODS.L_PUBL_RATE AS R ON R.DATA_DATE = IS_DATE AND R.BASIC_CCY = TRIM(B.CURR_CD) AND R.FORWARD_CCY = 'CNY' AND R.DATA_DATE = IS_DATE
- INNER JOIN (SELECT T.CONTRACT_NUM, COUNT(DISTINCT SUBSTRING(T1.GUAR_TYP, 1, 1)) AS CN FROM SMTMODS.L_AGRE_GUA_RELATION AS T INNER JOIN SMTMODS.L_AGRE_GUARANTEE_CONTRACT AS T1 ON T.GUAR_CONTRACT_NUM = T1.GUAR_CONTRACT_NUM AND T1.GUAR_CONTRACT_STATUS = 'Y' AND T1.DATA_DATE = IS_DATE WHERE T.DATA_DATE = IS_DATE AND T.REL_STATUS = 'Y' GROUP BY T.CONTRACT_NUM) AS DBFS ON A.ACCT_NO = DBFS.CONTRACT_NUM
- INNER JOIN JS_205_CLYHCD_TMP2 AS DBW ON A.ACCT_NO = DBW.BUSINESSCODE
- INNER JOIN SMTMODS.L_AGRE_GUARANTEE_CONTRACT AS T1 ON T.GUAR_CONTRACT_NUM = T1.GUAR_CONTRACT_NUM AND T1.GUAR_CONTRACT_STATUS = 'Y' AND T1.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `T.CJRQ = IS_DATE`
2. `NOT EXISTS(SELECT 1 FROM PBOCD_JS_205_CLYHCD_SQ AS F WHERE F.CJRQ = VS_LAST_TEXT AND T.BILL_NUM = F.BILL_NUM)`
3. `TABLE_NAME = 'JS_205_YHCDFS'`
4. `PARTITION_NAME = 'P' || IS_DATE`
5. `LOAN.DATA_DATE BETWEEN CAST(ADD_MONTHS(STR_TO_DATE(IS_DATE, '%Y%m%d'), -1) + 1 AS TEXT) AND IS_DATE`
6. `LOAN.BALANCE = 0`
7. `A.DATA_DATE = IS_DATE`
8. `A.BALANCE = 0`
9. `A.TRAN_AMT > 0`
10. `A.TRAN_AMT * R.CCY_RATE > 0`
11. `A.ACCT_TYP IN ('111', '112')`
12. `CAST(A.BUSINESS_DT AS TEXT) = SUBSTRING(IS_DATE, 1, 6)`
13. `TRIM(T.CJRQ) = IS_DATE`

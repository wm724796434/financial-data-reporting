# 存贷款客户核对_JS_201_HDACLHLDK — 取数范围

> 来源程序: `bsp_sp_js_201_hdaclhldk.prc`

## 数据源表（FROM）

- L_PUBL_ORG_BRA_TMP AS OB
- SMTMODS.L_ACCT_INTERNET_LOAN AS T1
- SMTMODS.L_CUST_C AS LC
- SMTMODS.L_CUST_COOP_AGEN AS C
- SMTMODS.L_CUST_P AS P

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_INTERNET_LOAN AS T1 ON T.LOAN_NUM = T1.LOAN_NUM AND T1.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_COOP_AGEN AS C ON T1.COOP_CUST_ID = C.COOP_CUST_ID AND C.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS P ON T.CUST_ID = P.CUST_ID AND P.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS LC ON LC.DATA_DATE = IS_DATE AND T.CUST_ID = LC.CUST_ID AND LC.CUST_TYP <> '3'

## 筛选条件（WHERE）

1. `T.DATA_DATE = IS_DATE`
2. `T.INTERNET_LOAN_FLG = 'Y'`
3. `T.LOAN_ACCT_BAL > 0`
4. `T.LOAN_STOCKEN_DATE IS NULL`
5. `T.CJRQ = IS_DATE`
6. `LC.CUST_ID IS NULL`

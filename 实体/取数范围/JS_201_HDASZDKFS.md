# JS_201_HDASZDKFS — 取数范围

> 来源程序: `bsp_sp_js_201_hdaszdkfs.prc`

## 数据源表（FROM）

- L_PUBL_ORG_BRA_TMP AS OB
- SMTMODS.L_ACCT_LOAN AS A1
- SMTMODS.L_AGRE_LOAN_CONTRACT AS D
- SMTMODS.L_CUST_ALL AS A2
- SMTMODS.L_CUST_C AS C
- SMTMODS.L_PUBL_RATE AS B
- SMTMODS.L_PUBL_RATE AS U

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_ACCT_LOAN AS A1 ON A1.DATA_DATE = IS_DATE AND A.ACCT_NUM || A.DRAFT_RNG = A1.ACCT_NUM || A1.DRAFT_RNG AND SUBSTRING(A1.ITEM_CD, 1, 6) IN ('130101', '130104')
- INNER JOIN SMTMODS.L_CUST_C AS C ON A.CUST_ID = C.CUST_ID AND C.DATA_DATE = IS_DATE AND C.CUST_TYP <> '3'
- INNER JOIN SMTMODS.L_AGRE_LOAN_CONTRACT AS D ON D.DATA_DATE = IS_DATE AND A.ACCT_NUM = D.CONTRACT_NUM AND SUBSTRING(D.DIGITAL_ECONOMY_INDUSTRY, 1, 2) IN ('01', '02', '03', '04')
- INNER JOIN SMTMODS.L_PUBL_RATE AS B ON B.DATA_DATE = IS_DATE AND B.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND A.CURR_CD = B.BASIC_CCY AND B.FORWARD_CCY = 'CNY'
- INNER JOIN SMTMODS.L_CUST_ALL AS A2 ON A2.DATA_DATE = IS_DATE AND A2.CUST_ID = T.CUST_ID
- INNER JOIN SMTMODS.L_PUBL_RATE AS U ON U.DATA_DATE = IS_DATE AND U.CCY_DATE = STR_TO_DATE(IS_DATE, '%Y%m%d') AND U.BASIC_CCY = T.CURR_CD AND U.FORWARD_CCY = 'CNY'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `A1.FUND_USE_LOC_CD = 'I'`
2. `A1.DATA_DATE = IS_DATE`
3. `LENGTHB(A1.ACCT_NUM) < 360`
4. `SUBSTRING(A.ITEM_CD, 1, 6) IN ('130102', '130105')`
5. `CAST(A.DRAWDOWN_DT AS TEXT) = SUBSTRING(IS_DATE, 1, 6)`
6. `SUBSTRING(A1.DIGITAL_ECONOMY_INDUSTRY, 1, 2) IN ('01', '02', '03', '04')`
7. `FLAG = 'SZDK'`
8. `FIELD_TYPE = 'T01'`
9. `FIELD_TYPE IN ('T02', 'T03', 'T04', 'T05', 'T06')`

# 担保物信息_JS_201_DBWXX — 取数范围

> 来源程序: `bsp_sp_js_201_dbwxx.prc`

## 数据源表（FROM）

- (SELECT ACCT_NUM, LOAN_NUM, DEPARTMENTD, ROW_NUMBER() OVER (PARTITION BY acct_num ORDER BY loan_num DESC NULLS FIRST) AS rn FROM SMTMODS.L_ACCT_LOAN WHERE data_date = IS_DATE) AS T9
- L_CODE_DICTIONARY AS T8
- L_PUBL_ORG_BRA_TMP AS OB
- L_PUBL_ORG_BRA_TMP AS OB2
- SMTMODS.L_PUBL_RATE AS T5

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_PUBL_RATE AS T5 ON T.PGBZ = T5.BASIC_CCY AND T5.FORWARD_CCY = 'CNY' AND T5.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS T8 ON TRIM(T.COLLATERAL_TYPE) = TRIM(T8.L_CODE) AND T8.CODE_CLMN_NAME = 'COLL_TYP'
- INNER JOIN (SELECT ACCT_NUM, LOAN_NUM, DEPARTMENTD, ROW_NUMBER() OVER (PARTITION BY acct_num ORDER BY loan_num DESC NULLS FIRST) AS rn FROM SMTMODS.L_ACCT_LOAN WHERE data_date = IS_DATE) AS T9 ON T.BUSINESSCODE = T9.ACCT_NUM AND T9.RN = '1'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.NBJGH AND OB.DATA_DATE = IS_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB2 ON OB.UP_ORG_NUM = OB2.ORG_NUM AND OB.DATA_DATE = IS_DATE AND OB2.DATA_DATE = IS_DATE

## 筛选条件（WHERE）

1. `T.DATA_DATE = IS_DATE`
2. `T.CANCEL_FLG = 'Y'`
3. `T.LOAN_STOCKEN_DATE IS NULL`
4. `TABLE_NAME = 'JS_201_DBWXX'`
5. `PARTITION_NAME = 'JS_201_DBWXX_' || IS_DATE`
6. `T.COLLATERAL_SERIAL_NUM <> '1100001352'`
7. `T.CJRQ = IS_DATE`

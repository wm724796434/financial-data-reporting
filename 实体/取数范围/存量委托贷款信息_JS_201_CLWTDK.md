# 存量委托贷款信息_JS_201_CLWTDK — 取数范围

> 来源程序: `bsp_sp_js_201_clwtdk.prc`

## 数据源表（FROM）

- CUST_WT_TEMP01 AS CW
- DBFS_TMP AS TP7
- JS_201_CLWTDK_TEMP03 AS T3
- L_CODE_DICTIONARY AS CD1
- L_CODE_DICTIONARY AS CD2
- L_CODE_DICTIONARY AS CD3
- L_CODE_DICTIONARY AS D4
- L_CUST_C_tmp AS G
- L_PUBL_ORG_BRA_TMP AS OB
- PBOCD_JS_201_CLWTDK_SQ AS BK
- SMTMODS.L_ACCT_LOAN_ENTRUST AS E
- SMTMODS.L_ACCT_LOAN_ENTRUST AS K
- SMTMODS.L_CUST_ALL AS B
- SMTMODS.L_CUST_C AS LC
- SMTMODS.L_CUST_P AS LP
- SMTMODS.L_PUBL_RATE AS R
- SMTMODS.v_pub_idx_dk_zqdqrjj AS ZQ

## 关联条件（JOIN）

- INNER JOIN SMTMODS.L_CUST_C AS LC ON T.CUST_ID = LC.CUST_ID AND LC.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS B ON T.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS LP ON T.CUST_ID = LP.CUST_ID AND LP.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS CD1 ON LC.id_type = CD1.L_CODE AND CD1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD2 ON LP.ID_TYPE = CD2.L_CODE AND CD2.CODE_CLMN_NAME = 'BZR_ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD3 ON LC.CORP_HOLD_TYPE = CD3.L_CODE AND CD3.CODE_CLMN_NAME = 'CORP_HOLD_TYPE'
- INNER JOIN L_CUST_C_tmp AS G ON T.CUST_ID = G.CUST_ID AND G.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS D4 ON TRIM(G.DEPT_TYPE) = D4.L_CODE AND D4.CODE_CLMN_NAME = 'DEPT_TYPE'
- INNER JOIN SMTMODS.L_ACCT_LOAN_ENTRUST AS K ON T.LOAN_NUM = K.LOAN_NUM AND K.DATA_DATE = IS_DATE
- INNER JOIN L_CUST_C_tmp AS G ON K.TRUSTOR_ID = G.CUST_ID AND G.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_P AS LP ON K.TRUSTOR_ID = LP.CUST_ID AND LP.DATA_DATE = IS_DATE
- INNER JOIN L_CODE_DICTIONARY AS CD1 ON G.ID_TYPE = CD1.L_CODE AND CD1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD2 ON LP.ID_TYPE = CD2.L_CODE AND CD2.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN L_CODE_DICTIONARY AS CD3 ON G.CORP_HOLD_TYPE = CD3.L_CODE AND CD3.CODE_CLMN_NAME = 'CORP_HOLD_TYPE'
- INNER JOIN JS_201_CLWTDK_TEMP03 AS T3 ON T.LOAN_NUM = T3.LOAN_NUM
- INNER JOIN SMTMODS.L_PUBL_RATE AS R ON R.DATA_DATE = IS_DATE AND R.BASIC_CCY = T.CURR_CD AND R.FORWARD_CCY = 'CNY' AND R.DATA_DATE = IS_DATE
- INNER JOIN DBFS_TMP AS TP7 ON T.LOAN_NUM = TP7.LOAN_NUM
- INNER JOIN CUST_WT_TEMP01 AS CW ON T.LOAN_NUM = CW.LOAN_NUM
- INNER JOIN SMTMODS.L_ACCT_LOAN_ENTRUST AS E ON T.LOAN_NUM = E.LOAN_NUM AND E.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.v_pub_idx_dk_zqdqrjj AS ZQ ON T.LOAN_NUM = ZQ.LOAN_NUM AND T.DATA_DATE = ZQ.DATA_DATE
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN PBOCD_JS_201_CLWTDK_SQ AS BK ON T.LOAN_NUM = BK.LOAN_NUM AND BK.CJRQ = VS_LAST_TEXT

## 筛选条件（WHERE）

1. `TABLE_NAME = 'JS_201_CLWTDK'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `T.DATA_DATE = IS_DATE`
4. `T.ITEM_CD LIKE '3020%'`
5. `NOT T.ITEM_CD IN ('30200201', '30200202')`
6. `T.CANCEL_FLG = 'N'`
7. `T.LOAN_STOCKEN_DATE IS NULL`
8. `TRIM(LC.Legal_Card_TYPE) = D4.L_CODE`
9. `D4.CODE_CLMN_NAME = 'ID_TYPE'`
10. `T.LOAN_ACCT_BAL > 0`
11. `TRIM(T.DATA_DATE) = IS_DATE`

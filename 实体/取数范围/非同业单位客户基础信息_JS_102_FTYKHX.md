# 非同业单位客户基础信息_JS_102_FTYKHX — 取数范围

> 来源程序: `bsp_sp_js_102_ftykhx.prc`

## 数据源表（FROM）

- (SELECT * FROM SMTMODS.L_ACCT_LOAN AS T WHERE t.data_date = IS_DATE AND (T.LOAN_ACCT_BAL > 0 OR DATE_TRUNC('MM', T.DRAWDOWN_DT) = DATE_TRUNC('MM', STR_TO_DATE(IS_DATE, 'yyyymmdd'))) AND (T.ITEM_CD LIKE '1303%' OR T.ITEM_CD LIKE '1305%') AND T.LOAN_STOCKEN_DATE IS NULL AND T.CANCEL_FLG = 'N') AS a
- (SELECT * FROM SMTMODS.L_ACCT_LOAN AS T WHERE t.data_date = IS_DATE AND T.ITEM_CD LIKE '1306%' AND T.CANCEL_FLG = 'N' AND T.LOAN_STOCKEN_DATE IS NULL) AS a
- (SELECT A.CUST_ID_NO, COUNT(*) AS CT FROM PBOCD_JS_102_FTYKHX_SQ AS A WHERE TRIM(A.DATA_DATE) = VS_LAST_TEXT GROUP BY A.CUST_ID_NO HAVING COUNT(*) > 1) AS C
- (SELECT CUST_ID, SUM(CONTRACT_AMT) AS CONTRACT_AMT FROM (SELECT A.CUST_ID, CASE WHEN A.CURR_CD = 'CNY' THEN A.CONTRACT_AMT ELSE A.CONTRACT_AMT * T1.CCY_RATE END AS CONTRACT_AMT FROM (SELECT CUST_ID, CONTRACT_AMT, CURR_CD FROM SMTMODS.L_AGRE_LOAN_CONTRACT WHERE DATA_DATE = IS_DATE) AS A LEFT
- (SELECT CUST_ID, SUM(LOAN_ACCT_BAL) AS LOAN_ACCT_BAL FROM (SELECT A.CUST_ID, CASE WHEN A.CURR_CD = 'CNY' THEN A.LOAN_ACCT_BAL ELSE A.LOAN_ACCT_BAL * T1.CCY_RATE END AS LOAN_ACCT_BAL FROM (SELECT CUST_ID, LOAN_ACCT_BAL, CURR_CD FROM SMTMODS.L_ACCT_LOAN WHERE DATA_DATE = IS_DATE AND ACCT_TYP NOT LIKE '90%') AS A LEFT
- (SELECT T.CUST_ID, SUM(T.FACILITY_AMT) AS FACILITY_AMT, SUM(T.USED_FACILITY_AMT) AS USED_FACILITY_AMT FROM SMTMODS.L_AGRE_CREDITLINE AS T WHERE T.DATA_DATE = IS_DATE AND T.FACILITY_STS = 'Y' GROUP BY T.CUST_ID) AS T1
- (SELECT T.CUST_ID, SUM(T.LOAN_ACCT_BAL) AS BAL FROM SMTMODS.L_ACCT_LOAN AS T WHERE T.DATA_DATE = IS_DATE AND T.CANCEL_FLG = 'N' AND T.LOAN_STOCKEN_DATE IS NULL AND T.LOAN_ACCT_BAL > 0 AND (T.ITEM_CD LIKE '1303%' OR T.ITEM_CD LIKE '1305%') GROUP BY T.CUST_ID) AS D
- FTY_SJKZR_02 AS F
- JS_102_FTYKHX_MAPPING AS M
- JS_102_FTYKHX_MAPPING AS M3
- JS_102_FTYKHX_TEMP01 AS T1
- JS_102_FTYKHX_TEMP02 AS T2
- JS_102_FTYKHX_TEMP03 AS T3
- JS_102_FTYKHX_TEMP04 AS HT
- JS_102_FTYKHX_TEMP09 AS T9
- L_CODE_DICTIONARY AS M1
- M_EAST_META_FIELD_SCOPE AS M2
- PBOCD_DATACORE.L_CODE_DICTIONARY AS D1
- PBOCD_JS_102_FTYKHX_SQ AS T1
- SMTMODS.L_ACCT_LOAN AS B
- SMTMODS.L_PUBL_RATE AS T1

## 关联条件（JOIN）

- INNER JOIN (SELECT * FROM SMTMODS.L_ACCT_LOAN AS T WHERE t.data_date = IS_DATE AND (T.LOAN_ACCT_BAL > 0 OR DATE_TRUNC('MM', T.DRAWDOWN_DT) = DATE_TRUNC('MM', STR_TO_DATE(IS_DATE, 'yyyymmdd'))) AND (T.ITEM_CD LIKE '1303%' OR T.ITEM_CD LIKE '1305%') AND T.LOAN_STOCKEN_DATE IS NULL AND T.CANCEL_FLG = 'N') AS a ON a.ACCT_NUM = T.CONTRACT_NUM
- INNER JOIN (SELECT * FROM SMTMODS.L_ACCT_LOAN AS T WHERE t.data_date = IS_DATE AND T.ITEM_CD LIKE '1306%' AND T.CANCEL_FLG = 'N' AND T.LOAN_STOCKEN_DATE IS NULL) AS a ON a.ACCT_NUM = T.CONTRACT_NUM
- INNER JOIN (SELECT CUST_ID, SUM(CONTRACT_AMT) AS CONTRACT_AMT FROM (SELECT A.CUST_ID, CASE WHEN A.CURR_CD = 'CNY' THEN A.CONTRACT_AMT ELSE A.CONTRACT_AMT * T1.CCY_RATE END AS CONTRACT_AMT FROM (SELECT CUST_ID, CONTRACT_AMT, CURR_CD FROM SMTMODS.L_AGRE_LOAN_CONTRACT WHERE DATA_DATE = IS_DATE) AS A LEFT JOIN SMTMODS.L_PUBL_RATE AS T1 ON A.CURR_CD = T1.BASIC_CCY AND T1.FORWARD_CCY = 'CNY' AND T1.DATA_DATE = IS_DATE) GROUP BY CUST_ID) AS LC ON T.CUST_ID = LC.CUST_ID
- INNER JOIN (SELECT CUST_ID, SUM(LOAN_ACCT_BAL) AS LOAN_ACCT_BAL FROM (SELECT A.CUST_ID, CASE WHEN A.CURR_CD = 'CNY' THEN A.LOAN_ACCT_BAL ELSE A.LOAN_ACCT_BAL * T1.CCY_RATE END AS LOAN_ACCT_BAL FROM (SELECT CUST_ID, LOAN_ACCT_BAL, CURR_CD FROM SMTMODS.L_ACCT_LOAN WHERE DATA_DATE = IS_DATE AND ACCT_TYP NOT LIKE '90%') AS A LEFT JOIN SMTMODS.L_PUBL_RATE AS T1 ON A.CURR_CD = T1.BASIC_CCY AND T1.FORWARD_CCY = 'CNY' AND T1.DATA_DATE = IS_DATE) GROUP BY CUST_ID) AS AL ON LC.CUST_ID = AL.CUST_ID
- INNER JOIN SMTMODS.L_PUBL_RATE AS T1 ON A.CURR_CD = T1.BASIC_CCY AND T1.FORWARD_CCY = 'CNY' AND T1.DATA_DATE = IS_DATE
- INNER JOIN (SELECT A.CUST_ID_NO, COUNT(*) AS CT FROM PBOCD_JS_102_FTYKHX_SQ AS A WHERE TRIM(A.DATA_DATE) = VS_LAST_TEXT GROUP BY A.CUST_ID_NO HAVING COUNT(*) > 1) AS C ON B.CUST_ID_NO = C.CUST_ID_NO
- INNER JOIN (SELECT T.CUST_ID, SUM(T.LOAN_ACCT_BAL) AS BAL FROM SMTMODS.L_ACCT_LOAN AS T WHERE T.DATA_DATE = IS_DATE AND T.CANCEL_FLG = 'N' AND T.LOAN_STOCKEN_DATE IS NULL AND T.LOAN_ACCT_BAL > 0 AND (T.ITEM_CD LIKE '1303%' OR T.ITEM_CD LIKE '1305%') GROUP BY T.CUST_ID) AS D ON B.CUST_ID = D.CUST_ID
- INNER JOIN L_CODE_DICTIONARY AS M1 ON TRIM(F.P_ID_TYPE) = M1.L_CODE AND M1.CODE_CLMN_NAME = 'ID_TYPE2'
- INNER JOIN (SELECT T.CUST_ID, SUM(T.FACILITY_AMT) AS FACILITY_AMT, SUM(T.USED_FACILITY_AMT) AS USED_FACILITY_AMT FROM SMTMODS.L_AGRE_CREDITLINE AS T WHERE T.DATA_DATE = IS_DATE AND T.FACILITY_STS = 'Y' GROUP BY T.CUST_ID) AS T1 ON T.CUST_ID = T1.CUST_ID
- INNER JOIN JS_102_FTYKHX_TEMP03 AS T3 ON T.CUST_ID = T3.CUST_ID
- INNER JOIN M_EAST_META_FIELD_SCOPE AS M2 ON SUBSTRB(TRIM(T.CORP_BUSINSESS_TYPE), 0, 3) = M2.PARAM_CODE AND M2.PARAM_TYPE = 'C0012'
- INNER JOIN JS_102_FTYKHX_TEMP04 AS HT ON T.CUST_ID = HT.CUST_ID
- INNER JOIN JS_102_FTYKHX_MAPPING AS M3 ON T.CUST_ID = M3.COD_CUST_ID
- INNER JOIN PBOCD_DATACORE.L_CODE_DICTIONARY AS D1 ON COALESCE(T.LEGAL_CARD_TYPE, T.ID_TYPE) = D1.L_CODE AND D1.CODE_CLMN_NAME = 'ID_TYPE'
- INNER JOIN JS_102_FTYKHX_TEMP01 AS T1 ON T.CUST_ID = T1.CUST_ID
- INNER JOIN JS_102_FTYKHX_MAPPING AS M ON T.CUST_ID = M.COD_CUST_ID
- INNER JOIN FTY_SJKZR_02 AS F ON T.CUST_ID = F.CUST_ID
- INNER JOIN SMTMODS.L_ACCT_LOAN AS B ON A.CUST_ID = B.CUST_ID AND B.DATA_DATE = IS_DATE AND B.DEPARTMENTD IN ('公司金融', '普惠金融')
- INNER JOIN SMTMODS.L_ACCT_LOAN AS B ON A.CUST_ID = B.CUST_ID AND A.DRAWDOWN_DT = B.DRAWDOWN_DT AND B.DATA_DATE = IS_DATE AND B.DEPARTMENTD IN ('公司金融', '普惠金融')
- INNER JOIN JS_102_FTYKHX_TEMP02 AS T2 ON T.CUST_ID_NO = T2.CUST_ID_NO AND (CASE WHEN T.NBJGH LIKE '51%' THEN '51' WHEN T.NBJGH LIKE '52%' THEN '52' WHEN T.NBJGH LIKE '53%' THEN '53' WHEN T.NBJGH LIKE '54%' THEN '54' WHEN T.NBJGH LIKE '55%' THEN '55' WHEN T.NBJGH LIKE '56%' THEN '56' WHEN T.NBJGH LIKE '57%' THEN '57' WHEN T.NBJGH LIKE '58%' THEN '58' WHEN T.NBJGH LIKE '59%' THEN '59' WHEN T.NBJGH LIKE '60%' THEN '60' ELSE '99' END) = T2.ORG_FLG
- INNER JOIN JS_102_FTYKHX_TEMP09 AS T9 ON T.CUST_ID = T9.CUST_ID
- INNER JOIN PBOCD_JS_102_FTYKHX_SQ AS T1 ON T.CUST_ID_NO = T1.CUST_ID_NO AND T1.CJRQ = VS_LAST_TEXT

## 筛选条件（WHERE）

1. `T.DATA_DATE = IS_DATE`
2. `T.ITEM_CD NOT LIKE '301%'`
3. `T.ITEM_CD NOT LIKE '302%'`
4. `T.ITEM_CD NOT LIKE '303%'`
5. `T.ITEM_CD NOT LIKE '304%'`
6. `A.DATA_DATE = IS_DATE`
7. `A.CANCEL_FLG = 'Y' OR NOT A.LOAN_STOCKEN_DATE IS NULL`
8. `NOT EXISTS(SELECT * FROM SMTMODS.L_ACCT_LOAN AS B WHERE B.DATA_DATE = IS_DATE AND (B.CANCEL_FLG = 'N' AND B.LOAN_STOCKEN_DATE IS NULL) AND A.CUST_ID = B.CUST_ID)`
9. `DATA_DATE = is_date`
10. `DATE_SOURCESD IN ('贸易融资', '普通贷款', '委托贷款')`
11. `t.acct_typ NOT LIKE '90' OR t.acct_typ IS NULL`
12. `T.DATA_DATE = is_date`
13. `T.DATE_SOURCESD = '保理'`
14. `NOT EXISTS(SELECT 1 FROM JS_102_FTYKHX_TEMP04 AS B WHERE T.CUST_ID = B.CUST_ID)`
15. `DATA_DATE = IS_DATE`
16. `TRIM(B.DATA_DATE) = VS_LAST_TEXT`
17. `F.DATA_DATE = IS_DATE`
18. `F.RALATION_TYP = '0'`
19. `NOT P_ID_TYPE IS NULL`
20. `NOT ID_NO IS NULL`
21. `ID_NO <> '无'`
22. `TABLE_NAME = 'JS_102_FTYKHX'`
23. `PARTITION_NAME = 'P' || IS_DATE`
24. `(T3.CUST_ID IS NULL OR T.CUST_ID IN ('8000692376', '8911498167'))`
25. `TRIM(A.DATA_DATE) = IS_DATE`
26. `TRIM(T.DATA_DATE) = IS_DATE`
27. `NOT T.CUST_ID_NO IS NULL`
28. `RN = 1`
29. `T.CJRQ = IS_DATE`
30. `A.RN = 1`

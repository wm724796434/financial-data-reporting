## 数据源表（FROM）

- (SELECT * FROM (SELECT A.*, ROW_NUMBER() OVER (PARTITION BY TYSHXYDM ORDER BY FINA_ORG_NAME NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE DATA_DATE = IS_DATE AND LEGAL_FLAG = 'Y' AND NOT TYSHXYDM IS NULL AND TYSHXYDM <> '000000000000000000' AND FINA_ORG_NAME NOT LIKE '%存托%' AND FINA_ORG_NAME NOT LIKE '%资管%' AND FINA_ORG_NAME NOT LIKE '%禁用%') AS A WHERE A.RN = 1) AS B
- (SELECT * FROM (SELECT A.CUST_ID, A.FINA_ORG_CODE, A.FINA_ORG_NAME, B.FINA_ORG_NAME AS FINA_ORG_NAME_FR, B.LEGAL_TYSHXYDM AS LEGAL_TYSHXYDM_FR, ROW_NUMBER() OVER (PARTITION BY A.FINA_ORG_NAME, A.FINA_ORG_CODE ORDER BY A.FINA_ORG_NAME NULLS LAST) AS RN FROM (SELECT * FROM (SELECT CUST_ID, FINA_ORG_CODE, FINA_ORG_NAME, TYSHXYDM, LEGAL_TYSHXYDM, ROW_NUMBER() OVER (PARTITION BY FINA_ORG_NAME, FINA_ORG_CODE ORDER BY TYSHXYDM NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE A.DATA_DATE = IS_DATE) AS A WHERE A.RN = 1) AS A LEFT
- (SELECT * FROM SMTMODS.L_ACCT_FUND_MMFUND AS T1 WHERE T1.DATA_DATE = IS_DATE AND SUBSTRING(T1.ACCT_TYP, 1, 3) IN ('202', '205', '105', '102')) AS T1
- (SELECT ACCT_NUM, COLL_SUBJECT_TYPE FROM (SELECT ACCT_NUM, COLL_SUBJECT_TYPE, ROW_NUMBER() OVER (PARTITION BY ACCT_NUM ORDER BY COLL_SUBJECT_TYPE NULLS LAST) AS RN FROM SMTMODS.L_AGRE_REPURCHASE_GUARANTY_INFO WHERE DATA_DATE = IS_DATE) AS A WHERE A.RN = 1) AS A
- (SELECT ACCT_NUM, DEAL_ACCT_NUM, MAIN_ORG_CN_NAME, REF_NUM, CUST_ID, CUST_SHORT_NAME, BALANCE, VALUE_DATE, END_DT, CONTRACT_AMT, GL_ITEM_CODE, IS_PRODUCT, CFETSMEMBERID FROM SMTMODS.L_ACCT_FUND_REPURCHASE WHERE DATA_DATE = IS_DATE AND GL_ITEM_CODE IN ('111101', '211101')) AS M
- (SELECT DISTINCT ECIF_CUST_ID AS ECIF_CUST_ID, LEGAL_TYSHXYDM, Data_Date FROM smtmods.L_CUST_BILL_TY WHERE NOT LEGAL_TYSHXYDM IS NULL AND NOT ECIF_CUST_ID IS NULL) AS FR
- L_PUBL_ORG_BRA_TMP AS OB
- SMTMODS.L_CUST_ALL AS AA
- SMTMODS.L_CUST_C AS BB
- SMTMODS.L_CUST_C AS T1
- SMTMODS.L_PUBL_RATE AS C

## 关联条件（JOIN）

- INNER JOIN (SELECT * FROM SMTMODS.L_ACCT_FUND_MMFUND AS T1 WHERE T1.DATA_DATE = IS_DATE AND SUBSTRING(T1.ACCT_TYP, 1, 3) IN ('202', '205', '105', '102')) AS T1 ON T.CONTRACT_NUM = T1.ACCT_NUM
- INNER JOIN SMTMODS.L_PUBL_RATE AS C ON C.DATA_DATE = IS_DATE AND C.BASIC_CCY = T1.CURR_CD AND C.FORWARD_CCY = 'CNY'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T1.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_ALL AS AA ON T1.CUST_ID = AA.CUST_ID AND AA.DATA_DATE = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS BB ON AA.CUST_ID = BB.CUST_ID AND BB.DATA_DATE = IS_DATE
- INNER JOIN (SELECT ACCT_NUM, DEAL_ACCT_NUM, MAIN_ORG_CN_NAME, REF_NUM, CUST_ID, CUST_SHORT_NAME, BALANCE, VALUE_DATE, END_DT, CONTRACT_AMT, GL_ITEM_CODE, IS_PRODUCT, CFETSMEMBERID FROM SMTMODS.L_ACCT_FUND_REPURCHASE WHERE DATA_DATE = IS_DATE AND GL_ITEM_CODE IN ('111101', '211101')) AS M ON T.CONTRACT_NUM = M.ACCT_NUM
- INNER JOIN (SELECT ACCT_NUM, COLL_SUBJECT_TYPE FROM (SELECT ACCT_NUM, COLL_SUBJECT_TYPE, ROW_NUMBER() OVER (PARTITION BY ACCT_NUM ORDER BY COLL_SUBJECT_TYPE NULLS LAST) AS RN FROM SMTMODS.L_AGRE_REPURCHASE_GUARANTY_INFO WHERE DATA_DATE = IS_DATE) AS A WHERE A.RN = 1) AS A ON A.ACCT_NUM = M.ACCT_NUM
- INNER JOIN SMTMODS.L_PUBL_RATE AS C ON C.DATA_DATE = IS_DATE AND C.BASIC_CCY = T.CURR_CD AND C.FORWARD_CCY = 'CNY'
- INNER JOIN L_PUBL_ORG_BRA_TMP AS OB ON OB.ORG_NUM = T.ORG_NUM AND OB.DATA_DATE = IS_DATE
- INNER JOIN (SELECT DISTINCT ECIF_CUST_ID AS ECIF_CUST_ID, LEGAL_TYSHXYDM, Data_Date FROM smtmods.L_CUST_BILL_TY WHERE NOT LEGAL_TYSHXYDM IS NULL AND NOT ECIF_CUST_ID IS NULL) AS FR ON M.CUST_ID = FR.ECIF_CUST_ID AND FR.Data_Date = IS_DATE
- INNER JOIN SMTMODS.L_CUST_C AS T1 ON M.CUST_ID = T1.CUST_ID AND T1.DATA_DATE = IS_DATE
- INNER JOIN (SELECT * FROM (SELECT A.CUST_ID, A.FINA_ORG_CODE, A.FINA_ORG_NAME, B.FINA_ORG_NAME AS FINA_ORG_NAME_FR, B.LEGAL_TYSHXYDM AS LEGAL_TYSHXYDM_FR, ROW_NUMBER() OVER (PARTITION BY A.FINA_ORG_NAME, A.FINA_ORG_CODE ORDER BY A.FINA_ORG_NAME NULLS LAST) AS RN FROM (SELECT * FROM (SELECT CUST_ID, FINA_ORG_CODE, FINA_ORG_NAME, TYSHXYDM, LEGAL_TYSHXYDM, ROW_NUMBER() OVER (PARTITION BY FINA_ORG_NAME, FINA_ORG_CODE ORDER BY TYSHXYDM NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE A.DATA_DATE = IS_DATE) AS A WHERE A.RN = 1) AS A LEFT JOIN (SELECT * FROM (SELECT A.*, ROW_NUMBER() OVER (PARTITION BY TYSHXYDM ORDER BY FINA_ORG_NAME NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE DATA_DATE = IS_DATE AND LEGAL_FLAG = 'Y' AND NOT TYSHXYDM IS NULL AND TYSHXYDM <> '000000000000000000' AND FINA_ORG_NAME NOT LIKE '%存托%' AND FINA_ORG_NAME NOT LIKE '%资管%' AND FINA_ORG_NAME NOT LIKE '%禁用%') AS A WHERE A.RN = 1) AS B ON A.LEGAL_TYSHXYDM = B.TYSHXYDM) WHERE RN = 1) AS FR ON T.CUST_ID = FR.FINA_ORG_CODE
- INNER JOIN (SELECT * FROM (SELECT A.*, ROW_NUMBER() OVER (PARTITION BY TYSHXYDM ORDER BY FINA_ORG_NAME NULLS LAST) AS RN FROM SMTMODS.L_CUST_BILL_TY AS A WHERE DATA_DATE = IS_DATE AND LEGAL_FLAG = 'Y' AND NOT TYSHXYDM IS NULL AND TYSHXYDM <> '000000000000000000' AND FINA_ORG_NAME NOT LIKE '%存托%' AND FINA_ORG_NAME NOT LIKE '%资管%' AND FINA_ORG_NAME NOT LIKE '%禁用%') AS A WHERE A.RN = 1) AS B ON A.LEGAL_TYSHXYDM = B.TYSHXYDM

## 筛选条件（WHERE）

1. `TABLE_NAME = 'PBOCD_JS_201_TYJDFS'`
2. `PARTITION_NAME = 'P' || IS_DATE`
3. `T.CJRQ = IS_DATE`
4. `NOT EXISTS(SELECT 1 FROM PBOCD_JS_201_CLTYJD_SQ AS F WHERE F.CJRQ = VS_LAST_TEXT AND T.CONTRACT_CODE = F.CONTRACT_CODE AND F.CURR_CODE <> 'CNY')`
5. `T.CURR_CODE <> 'CNY'`
6. `T1.ORG_NUM <> '009801'`
7. `SUBSTRING(CAST(TRAN_DT AS TEXT), 1, 6) = SUBSTRING(IS_DATE, 1, 6)`
8. `ITEM_CD IN ('111101', '211101')`
9. `RN = 1`
